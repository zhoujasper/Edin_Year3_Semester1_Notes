\subsection{Supremum and Infimum \& Set}

\textbf{Completeness Axiom}: If $\exists$ UP $\Rightarrow$ LUB (sup) ; If $\exists$ LB $\Rightarrow$ GLB (inf). \hspace{20pt} \textbf{Archimedean Property}: $\forall x \in \mathbb{R}^+ \Rightarrow \exists n\in \mathbb{N} \ s.t. \ n>x.$

\textbf{Countable Set}: A set $E$ is countable if $\exists$ bijection $f:\mathbb{N}\to E$. \ \ \ \ \ \ \ \ \ ps: $[0,1]$ is uncountable.

$\cdot$ \textbf{Property}: Let $(E_n)_{n\in\mathbb{N}}$ be countable sets. \ \ $\Rightarrow$ \ \ $^1\bigcup E_n$ is countable. (Can infinite) \ \ $^2\prod E_n$ is countable. (Finite) \ \ $^3[0,1]$ uncountable.

\ex\textbf{Dense Set}: Set $A$ is dense in $\mathbb{R}$ iff $\forall x\in\mathbb{R},\forall\varepsilon>0 \  \ \exists a\in A$ s.t. $|x-a|<\varepsilon$ \ \ \ \ \ \ {\scriptsize e.g.\ex $\mathbb{Q}$ is dense in $\mathbb{R}$}

\ex\textbf{Power Set $\mathcal{P}$}: $P(E)$ is the set of all subsets of $E$. \ \ \ \ \ \ \ \ \ ps: $\mathcal{P}(\mathbb{R})$ is the set of all subsets of $\mathbb{R}$.

\textbf{Property of Sup , Inf}: {\footnotesize $\inf(-f_n)=-\sup(f_n)$ \ \ || \ \ $\limsup(-f_n)=-\liminf(f_n)$ \ \ || \ \ If $g$ is fixed, $\sup(f_n+g)=\sup(f_n)+g$ \ \ ps:对$\inf,\limsup,\liminf$也成立}


\subsection{Nested Intervals}

\textbf{Def of $\lambda(I)$}: Let $I$ be an interval. $\lambda(I):=\sup I - \inf I$ \ \ || \ \ $\lambda((a,b))=\lambda([a,b))=\lambda((a,b])=\lambda([a,b])=b-a$.

\textbf{Nested Intervals}: Sequence $(I_n)_{n\in\mathbb{N}}$ is nested if $I_n\supseteq I_{n+1}$ ($\downarrow$) for all $n\in\mathbb{N}$.

\textbf{Nested Interval Theorem}: If $(I_n)_{n\in\mathbb{N}}$ is nested \textit{closed bounded} interval sequence $\Rightarrow$ $E:=\bigcap I_n\ne\emptyset$.

$\cdot$ \textbf{property}: $^1$If $\lim\lambda(I_n)=0\Rightarrow E=a$ \ \ $^2$\ex If $I_n=[a_n,b_n]$ \ \ $\Rightarrow$ \ \ $E=[\sup a_n,\inf b_n]$


\subsection{Sequence, Cauchy \& Bolzano Weierstrass Theorem}

\textbf{Useful Seq Limit}: $\lim \frac{1}{n^p}=0,p>1$ \ ; \ $\lim a^n=0,|a|<1$ \ ; \ $\lim n^\frac{1}{n}=1$ \ ; \ $\lim a^\frac{1}{n}=1,a>0$

\textbf{Operation in Seq Limit}: \textit{Squeeze Theorem} ; \textit{Monotone Convergence Theorem (MCT)} ; $\lim \alpha x_n=\alpha\lim x_n$

$\cdot$ $\lim (x_n + y_n)=\lim x_n + \lim y_n$ ; $\lim x_n \cdot y_n=\lim x_n \cdot \lim y_n$ ; $\lim \frac{x_n}{y_m}=\frac{\lim x_n}{\lim y_n},y_n \ne 0$

\textbf{Comparison Theorem (For Seq)}: If $\lim x_n,y_n$ exist, and $x_n\leq y_n$ for all $n\geq N$ then $\lim x_n\leq\lim y_n.$

\textbf{Cauchy Sequence}: Sequence $(a_n)_{n\in\mathbb{N}}$ is Cauchy iff $\forall\varepsilon>0,\exists N\in\mathbb{N} \ s.t. \ |a_n-a_m|<\varepsilon$ for all $m,n\geq N$. \ \ || \ \ \textbf{Convergent $\Leftrightarrow$ Cauchy}

\textbf{Bolzano-Weierstrass Theorem}: Every \textit{bounded sequence} has a \textit{convergent subsequence}.

\textbf{Def of limit superior and inferior}: $\limsup a_n:=\lim(sup_{k\geq n}a_k)$ \ \ || \ \ $\liminf a_n:=\lim(inf_{k\geq n}a_k)$

$\cdot$ \textbf{Property}: If $(a_n)_{n\in\mathbb{N}}$ bounded. \ \ $\Rightarrow$ \ \ \ex$^1\exists (a_{n_k})\to\lim\sup a_n$ {\tiny Hint:$\exists a_{n_k},\sup_{j\geq k}a_j-\frac{1}{k}\leq a_{n_k}\leq\sup_{j\geq k}a_j,\sup_{j\geq n_k}a_j\leq\sup_{j\geq k}a_j$} \ \ \ex$^2\exists (a_{n_k})\to\lim\inf a_n$

\hspace{157pt} \ex $^3\forall a_{n_k}\to a$ $\lim\inf a_n\leq a\leq\lim\sup a_n$

\textbf{Subsequence Theorem}: In each case, the subsequence can be taken to be \textit{monotonic}.
\begin{enumerate}[itemsep=-2pt, topsep=-2pt]
    \item $\exists a \in \mathbb{R} \ s.t. \ \forall \varepsilon > 0$, there are infinitely many $n \in \mathbb{N} \ \ s.t. \ |x_n - a|<\varepsilon$ {\footnotesize ($\exists$ 聚点a)} \ \ $\Leftrightarrow \ \ \exists$ subsequence of $(x_n) \ s.t. \ \lim\ x_{n_k}=a$
    \item $(x_n)_{n \in \mathbb{N}}$ is not bounded above (below) $\Leftrightarrow$ \ $\exists$ subsequence of $x_n \ s.t. \ \lim x_{n_k}=\infty \ (-\infty)$
    \item If $\lim x_n=L \Rightarrow \forall x_{n_k} \ , \lim x_{n_k}=L$
\end{enumerate}


\section{Series}

\textbf{Cauchy Criterion for Series}: $\sum^\infty_{k=1}a_k$ converges \ \ $\Leftrightarrow$ \ \ $\forall\varepsilon>0,\exists N\in\mathbb{N} \ s.t. \ \forall n\geq m\geq N \ |\sum^{n}_{k=m+1}a_k|=|a_{m+1}+\cdots+a_n|<\varepsilon$.

\textbf{p-Test}: $\sum \frac{1}{n^p}$ convergent iff $p>1$ \ \ \textbf{Root Test} \ \ \textbf{Ratio Test}$_{|n|^{1/n},|n+1/n|}$ \ \ \textbf{Geometric Series}: $\sum p^n$ converges iff $|p|<1$

\textbf{Divergence Test}: If $\lim a_n\ne 0 \Rightarrow \sum a_n$ diverges. \ \ (If $\sum a_n$ convergent $\Rightarrow \lim a_n=0.$)

\textbf{Comparison Test}: If $0<a_n<b_n$, $\sum b_n$ convergent $\Rightarrow \sum a_n$ also ; $\sum a_n$ divergent $\Rightarrow \sum b_n$ also.

$\cdot$ \textbf{Limit Comparison Test}: If $a_n,b_n\geq 0,\lim \frac{a_n}{b_n}=L>0 \ \Rightarrow$ If $L\in(0,\infty)$: $\sum a_n$ convergent iff $\sum b_n$ also.

\ \ \  If $L=0 : \sum b_n$ convergent $\Rightarrow \sum a_n$ also. \ \ $L=\infty : \sum b_n$ divergent $\Rightarrow \sum a_n$ also.

\textbf{Integral Test}: Let $f:[1,\infty)\rightarrow\mathbb{R}$ is 非负递减, $a_n=f(n)$. Then $\sum a_n$ converges iff $\int_{1}^{\infty}f(x)dx<\infty.$

\textbf{Absolutely Convergence}: $\sum a_n$ convergent absolutely iff $\sum |a_n|$ convergent. \ \ \textbf{If convergent abs $\Rightarrow$ convergent.}

\textbf{Alternating Series Test}: If $a_n$ decreasing, $a_n\geq 0,\lim a_n = 0$. Then $\sum (-1)^{n-1}a_n$ convergent.

\textbf{Cauchy's Condensation Test}: If $a_n\geq 0,a_n$ decreasing, $\Rightarrow \ [\ \sum a_n convergent \Leftrightarrow \sum 2^na_{2^n} \ also \ ]$

\textbf{Riemann Rearrangement Theorem}: If $\sum^\infty_{k=1}a_k$ is conditionally convergent. $\Rightarrow$ \ $\forall s\in\mathbb{R},\exists$ bijection $\sigma:\mathbb{N}\to\mathbb{N}$ \ s.t. $\sum_{n=1}^{\infty}a_{\sigma(n)}=s$.

\textbf{Rearrangement for Abs Series}: If $\sum_{n=1}^{\infty}a_n$ is absolutely convergent. $\Rightarrow$ \ $\forall$ bijection $\sigma:\mathbb{N}\to\mathbb{N}$ , $\sum_{n=1}^{\infty}a_{\sigma(n)}=\sum_{n=1}^{\infty}a_n$.


\section{Continuous \& Differentiable Functions Properties}

\textbf{Def of lim of func}: {\footnotesize \textbf{1}.$\forall\varepsilon>0,\exists\delta>0 \ s.t. 0<|x-x_0|<\delta, \Rightarrow |f(x)-L|<\varepsilon.$ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \textbf{2}.$\forall x_n\ne x_0 \ s.t. \lim (x_n)=x_0, then \ \lim f(x_n)=L$.}

\textbf{Def of cont of func}: {\footnotesize \textbf{1}.$\forall\varepsilon>0,\exists\delta>0 \ s.t. \ \forall x,|x-x_0|<\delta \ \Rightarrow |f(x)-f(x_0)|<\varepsilon.$ \ \ \textbf{2}.$\forall (x_n)_{n\in \mathbb{N}}$,$if \ \lim x_n = x_0 \ \Rightarrow \lim f(x_n)=f(x_0)$}

\textbf{Def of diffi}: {\scriptsize \textbf{1}.$f'(a)=\lim\limits_{x\to a}\frac{f(x)-f(a)}{x-a}$ \ \ \ \textbf{2}.$f'(a)=\lim\limits_{h\to 0}\frac{f(a+h)-f(a)}{h}$.}

\textbf{The Extreme Value Theorem (EVT)}: If $f:[a,b]\to\mathbb{R}$ is continuous, then $f$ has a max and min value.

\textbf{Intermediate Value Theorem (IVT)}: $f:[a,b]\to\mathbb{R}$ is continuo,$f(a)\ne f(b)$, $y\in[f(a),f(b)]$.$\Rightarrow \exists c\in(a,b) \ s.t. f(c)=y$

\textbf{IVT (for deriv) / (Darboux Theorem)}: If $f$ differentiable in $I$ $\Rightarrow\forall m\in(f'(a),f'(b))$ we have: $\exists c$ s.t. $f'(c)=m$

\textbf{Rolle's Theorem}: If $f$ continuous on $[a,b]$, differentiable on $(a,b)$ and $f(a)=f(b)$ \ \ Then: $\exists c\in(a,b)$ s.t. $f'(c)=0$

\textbf{Mean Value Theorem (MVT)}: Let $f,g$ continuous on $[a,b]$ and differentiable on $(a,b)$: \ \ \ {\scriptsize ps: a=b时依然成立}

$\cdot$ {\small \textbf{1}.$\exists c\in(a,b)$ s.t. $\frac{f(b)-f(a)}{b-a}=f'(c)$ \ \ \ \textbf{2}. $\exists c\in(a,b)$ s.t. $\frac{f(b)-f(a)}{g(b)-g(a)}=\frac{f'(c)}{g'(c)}$}

\textbf{Theorem}: If $f$ is 1-1 and continuous $\Rightarrow$ f is strictly monotone. \ \ \ and $f^{-1}$ is continuous and strictly monotone.

\textbf{Inverse Function Theorem}: Let $f$ differentiable in $I$ (open interval), and $\forall x,f'(x)\ne 0$:

$\cdot$ \textbf{1.} $f(I)$ is open interval. \ \ \ \ \ \textbf{2.} $f$ is invertible. \ \ \ \ \ \textbf{3.} $f^{-1}$ is differentiable and $(f^{-1})'(y)=\frac{1}{f'(f^{-1}(y))}$

\textbf{Second Deriv Test}: $f$ twice continuously differentiable in $I$,$f'(x_0)=0$: \textbf{1.} $f''(x_0)>0$ $(<0)$ local min (max) point.


\section{Pointwise \& Uniformly Convergence}

\subsection{Pointwise Convergence}

\textbf{Pointwise Convergence}: Let $(f_n)_{n\in\mathbb{N}}$ be a sequence of functions. $f_n:E\to\mathbb{R}$ converge pointwise to $f:E\to\mathbb{R}$ if: \ \ {\scriptsize 不加说明默认pointwise}

\begin{enumerate}[itemsep=-2pt, topsep=-2pt]
    \item \textbf{Version 1}: $\forall x\in E,\forall\varepsilon>0,\exists N_{\varepsilon,x} \ s.t. \ \forall n>N \ |f_n(x)-f(x)|<\varepsilon$
    \item \textbf{Version 2}: $\forall x\in E,\lim f_n(x)$ exists and $\lim f_n(x)=f(x)$
\end{enumerate}

\textbf{Pointwise Convergence NOT Preserve}: $^1$\textit{continuity}, \ \ $^2$\textit{differentiability}, \ \ $^3(\lim f_n)'\ne\lim f'_n$, \ \ $^4\int\lim f_n\ne\lim\int f_n$

$\cdot$ {\footnotesize 对应反例}: {\scriptsize $f_n:[0,1]\to\mathbb{R}:=x^n$} \ \ {\scriptsize $^2f_n:(-1,1)\to\mathbb{R}:=\sqrt{x^2+\frac{1}{n}}$} \ \ {\scriptsize $^3f_n:(-1,1)\to\mathbb{R}:=\frac{sin(nx)}{n}$} \ \ {\scriptsize $^4f_n:(0,1]\to\mathbb{R}:=\begin{cases} n,0<x\leq\frac{1}{n} \\ 0,\frac{1}{n}<x\leq1 \end{cases}$}


\subsection{Uniformly Convergence}

\textbf{Graph}: $\Gamma(f):=\{(x,f(x))| x\in E\}$ {\scriptsize ps:$f:E\to\mathbb{R}$}

\textbf{Vertical Neighbourhood} $\mathcal{N}(f,\varepsilon):=\{(x,y)| x\in E,|y-f(x)|<\varepsilon\}$ {\scriptsize ps:$f:E\to\mathbb{R}$}

\textbf{Uniformly Convergence}: $f_n:E\to\mathbb{R}$ converge uniformly iff: \hspace{50pt} || \ \ \textbf{Uniformly} $\Rightarrow$ \textbf{Pointwise}
\begin{enumerate}[itemsep=-2pt, topsep=-2pt]
    \item \textbf{Version 1}: $\forall\varepsilon>0,\exists N_{\varepsilon} \ s.t. \ \forall n>N,\forall x\in E,|f_n(x)-f(x)|<\varepsilon$
    \item \textbf{Version 2}: $\lim\left[ \sup_{x\in E}|f_n(x)-f(x)| \right]=0$
    \item \textbf{Version 3 | Uniformly Cauchy}: $\forall\varepsilon>0,\exists N_\varepsilon\in\mathbb{N} \ s.t. \ \forall x\in E, m,n>N \ |f_n(x)-f_m(x)|<\varepsilon$ \ \ \ || \ \ \ {\small \textbf{Converge Unif} \ $\Leftrightarrow$ \ \textbf{Unif Cauchy}}
    \item \textbf{Version 4 | Vertical Neighbourhood Test}: $\forall\varepsilon>0,\exists N\in\mathbb{N} \ s.t. \ \forall n>N,\Gamma(f_n)\subset\mathcal{N}(f,\varepsilon)$
\end{enumerate}

\textbf{Sequence Test}: If $f_n:E\to\mathbb{R}$ converge uniformly to $f$. \ \ $\Rightarrow$ \ \ If $(x_n)_{n\in\mathbb{N}}\in E,\lim|f_n(x_n)-f(x_n)|=0$ \ \ {\footnotesize 不能当作 $\forall x_n$}

\textbf{Properties of Uniformly Convergent}: Let $f_n,f,g:I\to\mathbb{R}$ Then: \ \ \ \ \ {\scriptsize NOT preserve: \ differentiable.}
\begin{enumerate}[itemsep=-2pt, topsep=-2pt]
    \item \textbf{Continuous}: {\footnotesize If $^1\lim f_n(x)=f(x)$ uniformly, $^2\forall f_n$ continuous at $x_0$ \ $\Rightarrow$ \ $f$ continuous at $x_0$.} \hspace{80pt} {\footnotesize $\Downarrow$ $\forall x\in I,(\lim f_n)'(x)=\lim f'_n(x)$}
    \item \textbf{Termwise differentiation}: {\footnotesize If $^1I$ is \textit{bounded open}; $^2\forall f_n$ diff, $\lim f'_n=g$ unif; \underline{$^3\exists x_0\in I \ s.t. \ f_n(x_0)$ converges}. \ \ $\Rightarrow$ \ \ \underline{$\lim f_n=f$ unif}; $f'$ diff and $f'=g$.}
    \item \textbf{Uniform Convergence Theorem}: {\scriptsize If $^1\lim f_n(x)=f(x)$ \textit{uniformly} on $E\in\mathcal{M},m(E)<\infty$, $^2$ $f_n\in\mathcal{L}^1(\mathbb{R})$ \ \ $\Rightarrow$ \ \ $^1f\in\mathcal{L}^1(E)$ \ \ $^2\lim \int_Ef_n(x)dx=\int_E f dx$}
    \item \ex If $f_n:[a,b]\to\mathbb{R}$,$g_n:[a,b]\to\mathbb{R}$ converge uniformly. \ \ $\Rightarrow$ \ \ $\frac{f_n}{g_n}$ converge uniformly also. \ \ {\footnotesize ps:$(a,b)$不行} \ \ \ \ \ \ {\tiny Hint: $f_ng-fg_n=f_ng-fg+fg-fg_n$}
\end{enumerate}

\textbf{Def of Series of Functions}: Let $f_n:E\to\mathbb{R}$; Def partial sum $s_n(x):=\sum_{k=1}^{n}f_k(x)$

{\scriptsize \textbf{1}. $\sum f_n$ converge pointwise iff $s_n$ converge pointwise. \ \ \textbf{2}. $\sum f_n$ converge uniformly iff $s_n$ converge uniformly. \ \ \textbf{3}.$\sum f_n$ converge absolutely (pointwise) iff $\sum|f_n|$ converge (pointwise).}

\textbf{Weierstrass M-Test}: {\small $f_n:E\to\mathbb{R}$, $\exists (M_n)_{n\in\mathbb{N}}\geq 0$ s.t. $^1\forall x\in E,\forall n\in\mathbb{N}:|f_n|<M_n$; $^2\sum M_n$ converges. \ \ $\Rightarrow$ \ \ $\sum f_n$ converge absolutely \& Uniformly.}


\section{Power Series}

\subsection{Power Series, Radius of Convergence, Continuity and Differentiability}

\textbf{Power Series}: A series from $\sum_{n=0}^{\infty}a_n(x-c)^n$ \ \ \ \textbf{Center}: $c$. \ \ \ \textbf{Coefficients}: $a_n$.

\textbf{Radius of Convergence (R)}: For a power series $\sum a_n(x-c)^n$, there are three situations for it's radius of convergence:

\begin{enumerate}[itemsep=-2pt, topsep=-2pt]
    \item $R=0$ if $\sum a_n(x-c)^n$ converges only $x=c$.
    \item $R=\infty$ if $\sum a_n(x-c)^n$ converges absolutely $\forall x\in\mathbb{R}$.
    \item $\exists R>0$ s.t. $\sum a_n(x-c)^n$ converges absolutely if $|x-c|<R$; diverges if $|x-c|>R$.
    \item \textbf{All-In-One}: $R=\sup\{r\geq0:(a_nr^n)_{n\in\mathbb{N}} \ is \ \underline{bounded}\}$
\end{enumerate}

\textbf{Properties | In Radius of Convergence}: Assume $f(x)=\sum a_n(x-c)^n$ with $R>0$.

\begin{enumerate}[itemsep=-2pt, topsep=-2pt]
    \item \textbf{Continuous}: $f(x)$ continuous on $(c-R,c+R)$.
    \item \textbf{Convergence}: $\forall 0<r<R, \ \ \ f(x)$ converges absolutely \& uniformly on $[c-r,c+r]$.
    \item \textbf{Preserve R}: $\sum^\infty_{n=0} a_n(x-c)^n$ and $\sum^\infty_{n=1} na_n(x-c)^{n-1}$ have same $R$. \ \ {\footnotesize ps: 即求导R不变}
    \item \textbf{Termwise differentiation}: $f(x)$ is differentiable on $(c-R,c+R)$ \ \ \ || \ \ \ $^1f'(x)=\sum^\infty_{n=1} na_n(x-c)^{n-1},x\in(c-R,c+R)$ \\
    $^2\forall 0<r<R, \ \ \ f'(x)$ converges absolutely and uniformly on $[c-r,c+r]$. \ \ \ \ \ \ \ \ {\footnotesize ps: 即可以对里面求导}
    \item \textbf{Coefficients \& Diff}: $^1f(x)$ is \textit{infinitely differentiable} on $(c-R,c+R)$ \ \  \ || \ \ \ $^2a_k=\frac{f^{(k)}(c)}{k!}$
    \item \textbf{Uniqueness}: If $\sum a_n(x-c)^n=\sum b_n(x-c)^n$ for $\forall x\in\mathbb{R},|x-c|<r_0$ \ \ $\Rightarrow$ \ \ $a_n=b_n$
\end{enumerate}


\subsection{Exponential Function \& Real Analytic Function}

\textbf{Exponential Function}: $E(x):=\sum_{n=0}^{\infty}\frac{x^n}{n!}$ with $R=\infty$\ \ \ \ \ \ {\scriptsize ps: $E(x)=\lim_{n\to\infty}(1+\frac{x}{n})^n$ {\tiny Hint: 证pointwise考虑$ln$,as $ln(1+\frac{x}{n})=\int^{1+\frac{x}{n}}_1\frac{1}{t}dt$,缩放,squeeze theorem.|uniformly考虑unif cont}}

\textbf{Real Analytic Function}: $f:I\to\mathbb{R}$ is real analytic \ iff \ $\exists (a_n)_{n\in\mathbb{N}}$ s.t. $f(x)=\sum_{n=0}^{\infty}a_n(x-c)^n$ for $\forall x\in I$ \ \ \ {\footnotesize i.e. Series converges pointwise on I.}

\textbf{Taylor Series}: Given $^1$\textit{Open Interval} $I$, $^2$\textit{infinitely differentiable} $f:I\to\mathbb{R}$. \ \ $\Rightarrow$ \ \ Taylor Series of $f$ at $c\in I$ $:=\sum_{n=0}^{\infty}\frac{f^{(n)}(c)}{n!}(x-c)^n$

$\cdot$ {\footnotesize Not all infinitely differentiable functions are real analytic: e.g.$f(x)=\{e^{-1/x^2},x>0;0,x<0$}

\textbf{Taylor's Theorem}: If $f:(a,b)\to\mathbb{R}$ is differentiable function at $x_0\in(a,b)$:
\begin{enumerate}[itemsep=-2pt, topsep=-2pt]
    \item $P_{n,f,x_0}(x):=f(x_0)+\sum_{k=1}^{n}\frac{f^{(k)}(x_0)}{k!}(x-x_0)^k$ \ \ \ (Taylor's polynomial of degree $n$ at $x_0$)
    \vspace{-4pt}
    \item If $f^{(n+1)}$ exists on $(a,b)$, then: $\exists \xi\in(x,x_0)$ s.t. $f(x)=P_{n,f,x_0}(x)+\frac{f^{(n+1)}(\xi)}{(n+1)!}(x-x_0)^{n+1}$
    \item Remainder $R_{n,f,x_0}:=f(x)-P_{n,f,x_0}$. \ \ \ || \ \ \ $f:I\to\mathbb{R}$ is real analytic iff $\lim R_{n,f,x_0}=0$ for $\forall x\in I$
\end{enumerate}

\begin{enumerate}[itemsep=-3pt, topsep=-2pt]
    \item \textbf{$\cup$ of open sets}: {\footnotesize $(E_n)$ are open sets (Infinite). \ \ $\Rightarrow$ \ \  $\bigcup E_n$ is open.}

    {\small $E$ is a nonempty open set \ $\Leftrightarrow$ \ $E$ = $\bigcup I_n$ , $I_n$ is open interval. (Unique)}
\end{enumerate}

\begin{enumerate}[itemsep=-3pt, topsep=-2pt]

    \item \textbf{$\cap$ of closed sets}: {\footnotesize $(F_n)$ are closed sets (Infinite). \ \ $\Rightarrow$ \ \  $\bigcap F_n$ is closed.}
    \item \textbf{Find Closed Set}: \\
    {\footnotesize If $f:\mathbb{R}\to\mathbb{R}$ continuous \ \ $\Rightarrow$ \ \ $\{x\in\mathbb{R}:f(x)\leq\lambda \ $ {\tiny (or $\geq\lambda$) \ (or $=\lambda$)} $ \ \}$ is closed.}
    \item \textbf{Sequentially Closed}: \\
    {\small If $F$ is closed set, \ \ $\Leftrightarrow$ \ \ $\forall(x_n)_{n\in\mathbb{N}}\in F$ convergent. $x=\lim x_n\in F$}
\end{enumerate}

\end{multicols}

\subsection{Lebesgue Outer Measure}

\textbf{Lebesgue Outer Measure ($m^*$)}: Can interpret $m^*$ as $m^*:\mathcal{P}(\mathbb{R})\to[0,\infty)$ \ \ \ \ \ \ \ \ \ {\scriptsize ps: 本质是寻找到最小的\textit{区间}可以覆盖$E$,用前面$\lambda$的定义来定义"长度"}

\begin{enumerate}[itemsep=-2pt, topsep=-2pt]
    \item \textbf{Version 1|Open Intervals}: $m^*(E):=\inf\left\{\sum_{n=1}^{\infty}\lambda(I_n) \ | \ E\subseteq\cup_{n=1}^{\infty}I_n,I_n \ is \ a \ open \ interval \right\}$
    \item \textbf{Version 2|Open Sets}: $m^*(E)=\inf\left\{\lambda(U),E\subseteq U,U \ is \ open \ set \right\}$ \ \ \ \ \ \ ps:$\lambda(U)=\sum \lambda(I_n)$
    \item \textbf{Remark}: If we cannot find $(I_n)_{n\in\mathbb{N}}$ s.t. $\sum\lambda(I_n)<\infty$ \ \ $\Rightarrow$ \ \ $m^*(E)=\infty$.
\end{enumerate}

\vspace{-12pt}
\begin{multicols}{2}

    $\cdot$ \textbf{Basic 3 Properties}: Let Set $E,F \ s.t. \ E\subseteq F\subset\mathbb{R}$.

\begin{enumerate}[itemsep=-2pt, topsep=-2pt]
    \item \textbf{Null Empty Set}: $m^*(\emptyset)=0$. \ \ \ || \ \ \ Null Set $E$: $m^*(E)=0$.
    \item \textbf{Monotonicity}: $m^*(E)\leq m^*(F)$.
    \item \textbf{Countable Subadditivity}: $m^*(\cup_{n=1}^{\infty}E_n)\leq\sum_{n=1}^{\infty}m^*(E_n)$.
\end{enumerate}

\columnbreak

\textbf{Useful Properties|By Def of inf}: Let $E$ be a set, $m^*(E)<\infty$

Then $\forall\varepsilon>0,\forall\delta>0$: $\exists (I_n)_{n\in\mathbb{N}}$ , $I_n$ open intervals s.t.: \ex

{\footnotesize $^1E\subseteq\bigcup I_n$ \ | \ $^2m^*(E)\leq\sum \lambda(I_n)<m^*(E)+\varepsilon$ \ | \ $^3I_n\cap E=\emptyset,\forall n$ \ | \ $^4\lambda(I_n)<\delta,\forall n$}

{\footnotesize $^5I_n\cap E\nsubseteq I_k,\forall n\ne k.$} \ {\scriptsize $\leftarrow$无覆盖项} \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ {\scriptsize 无无关项$\uparrow$} \ \ \ \ {\scriptsize ps: 也使用与是open set的.}

\subsection{Lebesgue Measurable}

\textbf{Lebesgue Measurable}: 

\begin{enumerate}[itemsep=-2pt, topsep=-2pt]
    \item \textbf{Version 1}: A set $E\subseteq\mathbb{R}$ is \textit{Lebesgue measurable} if $\forall\varepsilon>0,\exists$ open set $U\subseteq\mathbb{R}$ s.t. $E\subseteq U$ and $m^*(U\setminus E)<\varepsilon$ \ \ \ {\tiny ps: 被open set 从外"恰好"包住}
    \item \textbf{Version 2}: A set $E\subseteq\mathbb{R}$ is \textit{Lebesgue measurable} if $\forall\varepsilon>0,\exists$ closed set $U\subseteq\mathbb{R}$ s.t. $U\subseteq E$ and $m^*(E\setminus U)<\varepsilon$ \ \ \ {\tiny ps: 被closed set 从内"恰好"包住}
\end{enumerate}

$\cdot$ $\mathcal{M}:=$ collection of all Lebesgue measurable sets of $\mathbb{R}$.

\textbf{Property of Lebesgue Measurable Sets}:

\begin{enumerate}[itemsep=-2pt, topsep=-2pt]
    \item \textbf{Closed Set|Open Set|Interval}: Any closed set / open set / interval is Lebesgue measurable.
    \item \textbf{Union $\cup$|\ex Intersection $\cap$}: If $(E_n)_{n\in\mathbb{N}}$ is \textit{Lebesgue Measurable} \ \ $\Rightarrow$ \ \ Also: $^1E:=\cup^\infty_{n=1}E_n$ \ \ \ex$^2E:=\cap^\infty_{n=1}E_n$. \ \ {\tiny Hint: $(\bigcap E_n)^C=\bigcup E_n^C$}
    \item \textbf{Complement $E^C$}: if $E\subseteq\mathbb{R}$ is \textit{Lebesgue Measurable} \ \ $\Rightarrow$ \ \ $\mathbb{R}\setminus E$ is also \textit{Lebesgue Measurable}.
    \item \textbf{Measure Zero Set}: $m^*(E)=0$ \ \ $\Leftrightarrow$ \ \ $E$ Lebesgue measurable.
    \item \textbf{Translation Invariance}: $m(E)=m(x+E)$ \ \ \ \ \ \ || \ \ \ \ \ \ If $E$ is Lebesgue Measurable \ \ $\Rightarrow$ \ \ $x+E$ also.
    \item \textbf{Find Lebesgue Measurable Set}: A simple function $s$ \ \ $\Rightarrow$ \ \ $\{x\in\mathbb{R},s(x)>\lambda (<\lambda \ or =\lambda)\}$ is Lebesgue Measurable.
\end{enumerate}


\subsection{Lebesgue Measure}

\textbf{Lebesgue Measure}: \textit{Lebesgue Measure} is a function $m:\mathcal{M}\to[0,\infty)$ s.t. $m(E):=m^*(E),\forall m\in\mathcal{M}$ \ \ \ \ \ \ {\tiny ps: 原来是 $m^*:\mathcal{P}(\mathbb{R})\to[0,\infty)$, 现在是 $m:\mathcal{M}\to[0,\infty)$}

$\cdot$ \textbf{Property}: \textbf{$\star$ All properties of Lebesgue Outer Measure hold for Lebesgue Measure. $\star$}

\textbf{Countable Additivity for Measure}: If $(E_n)_{n\in\mathbb{N}}$ \textit{disjoint} \& \textit{Lebesgue Measurable}.\ \ $\Rightarrow$ \ \ $m(\cup^\infty_{n=1}E_n)=\sum^\infty_{n=1}m(E_n)$

    \vspace{-5pt}

\textbf{Characteristic Function}: $\chi_E(x):=\{ 1,x\in E \ \ ; \ \ 0,x\notin E$ \ \ \ ; \ \ \ $E\subset\mathbb{R}$

\textbf{Basic Integral Definitions}: $\int_{\mathbb{R}} \chi_I(x) \, dx = m(I) = \lambda(I)$ \ \ \ || \ \ \ $\int_{\mathbb{R}} c \chi_I(x) \, dx = c \lambda(I)$ \ \ \ || \ \ \ $\int_{\mathbb{R}}\sum c_j\chi_{I_j}(x)dx=\sum c_j\lambda(I_j) \ , \ c_j\geq0$

\textbf{Step Function}: $f:\mathbb{R}\to[0,\infty)$ is a (non-negative) step function if $f(x)=\sum c_j\chi_{I_j}(x)$ \ \ \ \ \ \ for some $c_j$ and $\chi_{I_j}$

$\cdot$ \textbf{Property}: If $f$ is a step function \ \ $\Rightarrow$ \ \ $\exists c_n,I_n$ s.t. $I_n$ \textit{disjoint} and $f(x)=\sum c_n\chi_{I_n}(x)$

\textbf{Defined of $\int fdx$}: $\int_{\mathbb{R}} f$ is defined iff: It's Lebesgue integrable or $f:\mathbb{R}\to[0,\infty)$ is Lebesuge Measurable.

\begin{enumerate}[itemsep=-2pt, topsep=-2pt]

\end{enumerate}
\textbf{Basic Properties of Integral|Positive}: Let $f,g:[0,\infty)\to\mathbb{R}$ be non-negative Lebesgue Measurable functions.
\begin{enumerate}[itemsep=-2pt, topsep=-2pt]
    \item \textbf{Monotonicity}: If $0\leq f(x)\leq g(x)$ \ \ $\Rightarrow$ \ \ $0\leq\int_{\mathbb{R}}f(x)dx\leq\int_{\mathbb{R}}g(x)dx$
    \item \textbf{Markov's Inequality}: For all $\lambda>0$ \ \ $\Rightarrow$ \ \ $m(\{x\in\mathbb{R}:f(x)>\lambda\})\leq\frac{1}{\lambda}\int_{\mathbb{R}}f(x)dx$ \ \ \ {\footnotesize 函数积分面积 > 其对应底下矩形面积}

%   \item \ex\textbf{Almost Everywhere|$f=0$}: If $E$ Lebesgue Measurable \& $\int_Ef(x)dx=0$ \ \ $\Rightarrow$ \ \ $m(\{x\in E:f(x)>0\})=0$ \ \ \ {\tiny Hint: $\{x|f(x)>0\}=\bigcup\{x|f(x)>\frac{1}{n}\}$}
\end{enumerate}

\textbf{MCT for Integral|Positive}: {\small $^1f_n:[0,\infty)\to\mathbb{R}$ non-decreasing ($\uparrow$) MF; \ $^2\lim f_n=f$ pointwise. \ \ $\Rightarrow$ \ \ $^1f$ is measurable; \ $^2\int_{\mathbb{R}}\lim f_ndx=\lim\int_{\mathbb{R}}f_n(x)dx$}

\textbf{More Properties of Integral|Positive}: Let $f,g:[0,\infty)\to\mathbb{R}$ be non-negative Lebesgue Measurable functions.
\begin{enumerate}[itemsep=-2pt, topsep=-2pt]
    \item \textbf{Linear Combination}: $\alpha,\beta>0$ \ \ $\Rightarrow$ \ \ $\int_{\mathbb{R}}\alpha f+\beta g =\alpha\int_{\mathbb{R}}f + \beta\int_{\mathbb{R}}g$
    \item \textbf{Union of Disjoint}: $E,F$ disjoint measurable sets \ \ $\Rightarrow$ \ \ $\int_{E\cup F}f=\int_Ef+\int_Ff$
    \item \textbf{'MCT for Int Series'}: {\small $^1f_n$ non-negative MF; \ $^2\sum f_n=f$ pointwise \ \ $\Rightarrow$ \ \ $^1f$ is measurable; \ $^2\int_{\mathbb{R}}\sum f_n=\sum\int_{\mathbb{R}}f_n$}
    \item \textbf{(Reverse) Fatou Lemma}: {\small $f_n$ non-negative MF \ \ $\Rightarrow$ \ \ $\int_{\mathbb{R}}\liminf f_n\leq\liminf\int_{\mathbb{R}}f_n \ \leq \ ( \ \limsup\int_{\mathbb{R}}f_n\leq\int_{\mathbb{R}}\limsup f_n \ )$}
\end{enumerate}


\subsection{Integrating Real-Valued Functions}

\textbf{Isolated Functions}: For a function $f$. We define that $f^+:=\max\{f,0\}$ \ , \ $f^-:=-\min\{f,0\}$ \ \ \ $\Rightarrow$ \ \ \ $f=f^+-f^-$ \ , \ $|f|=f^++f^-$.

$\cdot$ \textbf{Remark}: If $a<0$: $(af)^+=(-a)f^-$ and $(af)^-=-af^+$

\textbf{Lebesgue Integrable}: A MF $f:\mathbb{R}\to\mathbb{R}$ is Lebesgue Integrable \ iff \ $\int_{\mathbb{R}}|f(x)|dx<\infty$ \ \ $\Leftrightarrow$ \ \ $\int_{\mathbb{R}}f^+dx<\infty$ \& $\int_{\mathbb{R}}f^-dx<\infty$
\begin{enumerate}[itemsep=-2pt, topsep=-2pt]
    \item \textbf{Def of Integral}: If $f\in\mathcal{L}^1(\mathbb{R})$ \ \ $\Rightarrow$ \ \ $\int_{\mathbb{R}}f(x)dx:=\int_{\mathbb{R}}f^+(x)dx-\int_{\mathbb{R}}f^-(x)dx$
    \item \textbf{Extended Def}: If $f:E\to\mathbb{R}$ with $E$ Measurable Set \ s.t. \ $\tilde{f}:=f(x)\chi_E$ is Lebesgue Integrable \ \ $\Rightarrow$ \ \ $\int_Ef(x)dx:=\int_{\mathbb{R}}\tilde{f}(x)dx$    
    \item $\mathcal{L}^1(\mathbb{R}):=$ Collection of all Lebesgue Integrable functions on $\mathbb{R}$.
    \item $\mathcal{L}^1(E):=$ Collection of all Lebesgue Integrable functions on $E$.
\end{enumerate}

\textbf{Property of Integral}: Let $f,g\in\mathcal{L}^1(\mathbb{R})$. Then:
\begin{enumerate}[itemsep=-2pt, topsep=-2pt]
    \item \textbf{Triangular Inequality for Integrals}: $|\int_{\mathbb{R}}f(x)dx|\leq\int_{\mathbb{R}}|f(x)|dx$
    \item \textbf{Linearity of the Integral}: $\alpha f+\beta g\in\mathcal{L}^1(\mathbb{R})$ and $\int_{\mathbb{R}}\alpha f+\beta g=\alpha\int_{\mathbb{R}}f+\beta\int_{\mathbb{R}}g$ \ \ \ $\Rightarrow$ \ \ \ {\footnotesize \textbf{$\mathcal{L}^1(\mathbb{R})$ is a vector space}}
    \item \textbf{Union of Disjoint}: $E,F$ disjoint measurable sets \ \ $\Rightarrow$ \ \ $\int_{E\cup F}f=\int_Ef+\int_Ff$ (By Linearity of Integral)
    \item \textbf{'Comparison Test'}: If $^1f:\mathbb{R}\to\mathbb{R}$ measurable \ \ $^2\int_{\mathbb{R}}g(x)<\infty$ \ \ $^3|f|<g$ \ \ $\Rightarrow$ \ \ $f\in\mathcal{L}^1(\mathbb{R})$
    \item \textbf{Remark}: Can use p-test, Comparison test ... when we want to prove a function is Lebesgue Integrable.
\end{enumerate}

\textbf{Dominated Convergence Theorem (DCT)}: {\small $^1f_n$ is MF. \ \ $^2f_n\to f$ pointwise. \ \ $^3\exists g\in\mathcal{L}^1(\mathbb{R}) \ s.t. \ |f_n|\leq g$ \ \ $\Rightarrow$ \ \ $^1f\in\mathcal{L}^1(\mathbb{R})$ \ \ \& \ \ $\lim\int_{\mathbb{R}}f_n=\int_{\mathbb{R}}f$}

$\cdot$ \textbf{Uniform Convergence Theorem}: {\scriptsize If $^1\lim f_n(x)=f(x)$ \textit{uniformly} on $E\in\mathcal{M},m(E)<\infty$, $^2$ $f_n\in\mathcal{L}^1(\mathbb{R})$ \ \ $\Rightarrow$ \ \ $^1f\in\mathcal{L}^1(E)$ \ \ $^2\lim \int_Ef_n(x)dx=\int_E f dx$}

\textbf{Fundamental Theorem of Calculus (FTC)}: $f:[a,b]\to\mathbb{R}$ be \textit{continuous}. \ \ $\Rightarrow$ \ \ $^1F(x):=\int_{a}^{x}f(y)dy$ is differentiable \ \ $^2F'(x)=f(x)$

$\cdot$ \textbf{Corollary}: $f:[a,b]\to\mathbb{R}$ be \textit{continuously differentiable}. \ \ $\Rightarrow$ \ \ $\int_{a}^{b}f'(x)dx=f(b)-f(a)$

$\cdot$ \textbf{Remark}: $^1$ f is Lebesgue Integrable. (By Extreme value theorem) and. \ \ \ $^2$ {\small 需要闭区间(Closed)否则需要结合MCT或DCT极限逼近来用.}

\begin{enumerate}[itemsep=-2pt, topsep=-2pt]
    \item \textbf{Integration-By-Parts}: If $f,g:[a,b]\to\mathbb{R}$ be \textit{continuously differentiable}. \ \ $\Rightarrow$ \ \ $\int_{a}^{b}fg' \ dx=[fg]^b_a-\int_{a}^{b}f'g \ dx$
    \item \textbf{Integration-By-Substitution}: {\small If $f:[c,d]\to\mathbb{R}$ \textit{continuous} \& $\phi:[a,b]\to[c,d]$ \textit{continuous diff} \ \ $\Rightarrow$ \ \ $\int_{a}^{b}f(\phi)\phi'dx=\int_{\phi(a)}^{\phi(b)}f(y)dy$}
\end{enumerate}
