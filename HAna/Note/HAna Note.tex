\section{Series}

\textbf{Cauchy Criterion for Series}: $\sum^\infty_{k=1}a_k$ converges \ \ $\Leftrightarrow$ \ \ $\forall\varepsilon>0,\exists N\in\mathbb{N} \ s.t. \ \forall n\geq m\geq N \ |\sum^{n}_{k=m+1}a_k|=|a_{m+1}+\cdots+a_n|<\varepsilon$.

\textbf{p-Test}: $\sum \frac{1}{n^p}$ convergent iff $p>1$ \ \ \textbf{Root Test} \ \ \textbf{Ratio Test}$_{|n|^{1/n},|n+1/n|}$ \ \ \textbf{Geometric Series}: $\sum p^n$ converges iff $|p|<1$

\textbf{Divergence Test}: If $\lim a_n\ne 0 \Rightarrow \sum a_n$ diverges. \ \ (If $\sum a_n$ convergent $\Rightarrow \lim a_n=0.$)

\textbf{Comparison Test}: If $0<a_n<b_n$, $\sum b_n$ convergent $\Rightarrow \sum a_n$ also ; $\sum a_n$ divergent $\Rightarrow \sum b_n$ also.

$\cdot$ \textbf{Limit Comparison Test}: If $a_n,b_n\geq 0,\lim \frac{a_n}{b_n}=L>0 \ \Rightarrow$ If $L\in(0,\infty)$: $\sum a_n$ convergent iff $\sum b_n$ also.

\ \ \  If $L=0 : \sum b_n$ convergent $\Rightarrow \sum a_n$ also. \ \ $L=\infty : \sum b_n$ divergent $\Rightarrow \sum a_n$ also.

\textbf{Integral Test}: Let $f:[1,\infty)\rightarrow\mathbb{R}$ is 非负递减, $a_n=f(n)$. Then $\sum a_n$ converges iff $\int_{1}^{\infty}f(x)dx<\infty.$

\textbf{Absolutely Convergence}: $\sum a_n$ convergent absolutely iff $\sum |a_n|$ convergent. \ \ \textbf{If convergent abs $\Rightarrow$ convergent.}

\textbf{Alternating Series Test}: If $a_n$ decreasing, $a_n\geq 0,\lim a_n = 0$. Then $\sum (-1)^{n-1}a_n$ convergent.

\textbf{Cauchy's Condensation Test}: If $a_n\geq 0,a_n$ decreasing, $\Rightarrow \ [\ \sum a_n convergent \Leftrightarrow \sum 2^na_{2^n} \ also \ ]$

\textbf{Riemann Rearrangement Theorem}: If $\sum^\infty_{k=1}a_k$ is conditionally convergent. $\Rightarrow$ \ $\forall s\in\mathbb{R},\exists$ bijection $\sigma:\mathbb{N}\to\mathbb{N}$ \ s.t. $\sum_{n=1}^{\infty}a_{\sigma(n)}=s$.

\textbf{Rearrangement for Abs Series}: If $\sum_{n=1}^{\infty}a_n$ is absolutely convergent. $\Rightarrow$ \ $\forall$ bijection $\sigma:\mathbb{N}\to\mathbb{N}$ , $\sum_{n=1}^{\infty}a_{\sigma(n)}=\sum_{n=1}^{\infty}a_n$.

\textbf{Mean Value Theorem (MVT)}: Let $f,g$ continuous on $[a,b]$ and differentiable on $(a,b)$: \ \ \ {\scriptsize ps: a=b时依然成立}

$\cdot$ {\small \textbf{1}.$\exists c\in(a,b)$ s.t. $\frac{f(b)-f(a)}{b-a}=f'(c)$ \ \ \ \textbf{2}. $\exists c\in(a,b)$ s.t. $\frac{f(b)-f(a)}{g(b)-g(a)}=\frac{f'(c)}{g'(c)}$}

\textbf{Theorem}: If $f$ is 1-1 and continuous $\Rightarrow$ f is strictly monotone. \ \ \ and $f^{-1}$ is continuous and strictly monotone.

\subsection{Introduction, Radius of Convergence, Continuity and Differentiability of Power Series}

\textbf{Power Series}: A series from $\sum_{n=0}^{\infty}a_n(x-c)^n$ \ \ \ \textbf{Center}: $c$. \ \ \ \textbf{Coefficients}: $a_n$.

\textbf{Radius of Convergence (R)}: For a power series $\sum a_n(x-c)^n$, there are three situations for it's radius of convergence:
\begin{enumerate}[itemsep=-2pt, topsep=-2pt]
    \item $R=0$ if $\sum a_n(x-c)^n$ converges only $x=c$ \ \ \ || \ \ \ $R=\infty$ if $\sum a_n(x-c)^n$ converges absolutely $\forall x\in\mathbb{R}$
    \item $\exists R>0$ s.t. $\sum a_n(x-c)^n$ converges absolutely if $|x-c|<R$; diverges if $|x-c|>R$. \ \ \ || \ \ \ $R=\sup\{r\geq0:(a_nr^n)_{n\in\mathbb{N}} \ \underline{bounded}\}$
\end{enumerate}

\textbf{Theorem (Continuity, Uniformly and Differentiability)}: If $f(x)=\sum a_n(x-c)^n$ with $R>0$.
\begin{enumerate}[itemsep=-2pt, topsep=-2pt]
    \item $f(x)$ continuous on $(c-R,c+R)$
    \item $\forall 0<r<R,f(x)$ converges absolutely and uniformly on $[c-r,c+r]$
    \item $\sum^\infty_{n=0} a_n(x-c)^n$ and $\sum^\infty_{n=1} na_n(x-c)^{n-1}$ have same $R$.
    \item \textbf{Termwise differentiation}: $f(x)$ is differentiable on $(c-R,c+R)$ \ \ \ || \ \ \ $f'(x)=\sum^\infty_{n=1} na_n(x-c)^{n-1},x\in(c-R,c+R)$
    \\ $\forall 0<r<R,f(x),f'(x)$ converges absolutely and uniformly on $[c-r,c+r]$
    \item $f(x)$ is infinitely convergence on $(c-R,c+R)$ \ || \ $a_k=\frac{f^{(k)}(c)}{k!}$
\end{enumerate}


\subsection{The Exponential Function \& Real Analytic Function}

\textbf{The Exponential Function}: $E(x):=\sum_{n=0}^{\infty}\frac{x^n}{n!}$ with $R=\infty$ \ \ \ \ \ \ {\scriptsize ps: usually denoted by $exp(x)$ or $e^x$}

\textbf{Real Analytic Function}: $f:I\to\mathbb{R}$ is real analytic \ iff \ $\exists (a_n)_{n\in\mathbb{N}}$ s.t. $f(x)=\sum_{n=0}^{\infty}a_n(x-c)^n$ for $\forall x\in I$ \ \ \ {\footnotesize i.e. Series converges pointwise on I.}

\textbf{Taylor Series}: Given $^1$\textit{Open Interval} $I$, $^2$\textit{infinitely differentiable} $f:I\to\mathbb{R}$. \ \ $\Rightarrow$ \ \ Taylor Series of $f$ at $c\in I$ $:=\sum_{n=0}^{\infty}\frac{f^{(n)}(c)}{n!}(x-c)^n$

$\cdot$ {\footnotesize Not all infinitely differentiable functions are real analytic: e.g.$f(x)=\{e^{-1/x^2},x>0;0,x<0$}

\textbf{Taylor's Theorem}: If $f:(a,b)\to\mathbb{R}$ is differentiable function at $x_0\in(a,b)$:
\begin{enumerate}[itemsep=-2pt, topsep=-2pt]
    \item $P_{n,f,x_0}(x):=f(x_0)+\sum_{k=1}^{n}\frac{f^{(k)}(x_0)}{k!}(x-x_0)^k$ \ \ \ (Taylor's polynomial of degree $n$ at $x_0$)
    \vspace{-4pt}
    \item If $f^{(n+1)}$ exists on $(a,b)$, then: $\exists \xi\in(x,x_0)$ s.t. $f(x)=P_{n,f,x_0}(x)+\frac{f^{(n+1)}(\xi)}{(n+1)!}(x-x_0)^{n+1}$
    \item Remainder $R_{n,f,x_0}:=f(x)-P_{n,f,x_0}$. \ \ \ || \ \ \ $f:I\to\mathbb{R}$ is real analytic iff $\lim R_{n,f,x_0}=0$ for $\forall x\in I$
\end{enumerate}

\subsubsection{Definition of Integral and Step Function}

\textbf{Types of Integration \& Fundamental Theorem of Calculus}: 
\begin{enumerate}[itemsep=-2pt, topsep=-2pt]
    \item \textbf{Indefinite integrals}: Also called \textit{primitives} or \textit{anti-derivatives}. {\footnotesize ps: 不定积分}. || Definition: {\footnotesize 求导的逆运算} || e.g.$\int f'(x)dx=f(x)+c$
    \item \textbf{Definite integrals}: {\footnotesize ps: 定积分} || Definition: {\footnotesize 函数在指定区域下的面积} || $\int_{a}^{b}f'(x)dx=f(b)-f(a)$
    \item \textbf{Fundamental Theorem of Calculus}: $\int_{a}^{b}f'(x)dx=f(b)-f(a)$
\end{enumerate}

\textbf{Characteristic function $\chi_E$}: Given $E\subset\mathbb{R},\chi_E(x):=\{1,x\in E; \ 0,x\notin E.$ \ \ \ \ \ \ || {\footnotesize ps: $\lambda(I)=\sup I-\inf I$ \ \ \ | \ \ \ If $I=I_1\cup I_2,I_1\cap I_2=\emptyset\Rightarrow\lambda(I)=\lambda(I_1)+\lambda(I_2)$}

\textbf{Def of Simple Integral}: \textbf{1}.$\int_{\mathbb{R}}\chi_I(x)dx:=\lambda(I)$ \ \ \ \ \ \ \textbf{2}. $\int_{\mathbb{R}}c\chi_I(x)dx:=c\lambda(I)$ \ \ \ \ \ \ \textbf{3}. $\int_{\mathbb{R}}\sum_jc_j\chi_{I_j}(x)dx=\sum_jc_j\lambda(I_j)$

\textbf{Step Function}: \underline{non-negative} $f:\mathbb{R}\to[0,\infty)$ is a step function iff $f(x)=\sum_jc_j\chi_{I_j}(x)$ \ \ \ \ \ $\Rightarrow$ \ \ \ \ \ \ Def: $\int_{\mathbb{R}}f(x)dx=\sum_jc_j\lambda(I_j)$


\subsubsection{Open Set and Closed Set}

\textbf{Open Set}: A set $S$ is open iff $\forall x\in S,\exists r>0, \ s.t. \ (x-r,x+r)\subseteq S$ \ \ \ \ \ \ {\footnotesize ps: Collection of all open subsets of $\mathbb{R}$ is called \textit{the Euclidean topology}}

\begin{enumerate}[itemsep=-2pt, topsep=-2pt]
    \item \textbf{Structure of open sets}: If $S$ is a nonempty open set \ $\Leftrightarrow$ \ $S$ = \textit{unique} \underline{countable} \textit{union} of \underline{disjoint} \textit{open intervals}. {\tiny ps:即 open set = $\cup$ open intervals}
    \item \textbf{$\cup$ of open sets}: Let $\mathcal{U}$ be collection of some open sets. (Can Infinite) \ \ \ \ \ \ $\cup_{S\in\mathcal{U}}S$ is open.
    \item \textbf{$\cap$ of open sets}: Let $\mathcal{U}$ be collection of some open sets. (Finite) \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ $\cap_{S\in\mathcal{U}}S$ is open.
    \item \textbf{Find Open Set}: If $f:\mathbb{R}\to\mathbb{R}$ continuous \ \ $\Rightarrow$ \ \ $\{x\in\mathbb{R}:f(x)>\lambda \ (or <\lambda) \ \}$ is open.
    \item \textbf{Open Set $\to\cup$ Closed Set}: Any nonempty open set $U$ $\to$ countable union of \textit{disjoint closed} intervals. 
\end{enumerate}

\textbf{Closed Set}: A set $F\subseteq\mathbb{R}$ is \textit{closed} iff $\mathbb{R}\setminus F$ is open.

\begin{enumerate}[itemsep=-2pt, topsep=-2pt]
    \item \textbf{$\cup$ of closed sets}: Let $\mathcal{F}$ be collection of some closed sets. (Finite) \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ $\cup_{F\in\mathcal{F}}F$ is closed.
    \item \textbf{$\cap$ of closed sets}: Let $\mathcal{F}$ be collection of some closed sets. (Can Infinite) \ \ \ \ \ \ $\cap_{F\in\mathcal{F}}F$ is open.
    \item \textbf{Find Closed Set}: If $f:\underline{\mathbb{R}}\to\mathbb{R}$ continuous \ \ $\Rightarrow$ \ \ $\{x\in\mathbb{R}:f(x)\leq\lambda \ (or\geq\lambda) \ (or =\lambda) \ \}$ is closed.
    \item \textbf{Unbounded$\to\cup$Bounded}: Any closed set $F$ $\to$ countable union of \textit{bounded} closed intervals. \ \ \ {\footnotesize (Hint: $K_n=F\cap[-n,n]$)}
    \item \textbf{Sequencially Closed}: If $F$ is closed set, \ \ $\Leftrightarrow$ \ \ $\forall(x_n)_{n\in\mathbb{N}}\in F$ convergent. $x=\lim x_n\in F$
\end{enumerate}

\textbf{Property|Open \& Closed Set}: Let $E$ be open set, $F$ be closed set.
\begin{enumerate}[itemsep=-2pt, topsep=-2pt]
    \item $E \setminus F=F^C\cap E$ is open {\footnotesize (i.e. open - closed = open)} \ \ \ \ \ \ \ \ \ || \ \ \ \ \ \ \ \ \ $F\setminus E=E^C\cap F$ is open {\footnotesize (i.e. closed - open = closed)}
    \item Only $\emptyset$ and $\mathbb{R}$ are both open and closed.
    \item $\forall (E_n)_{n\in\mathbb{N}}$ \ \ $\Rightarrow$ \ \ $\exists$ disjoint $(F_n)_{n\in\mathbb{N}}$ s.t. $E_n\subseteq F_n$ and $\cup F_n=\cup E_n$. \ \ \ {\footnotesize (Hint: $F_n:=E_n\setminus\cup^{n-1}_{i=1}F_i$)}
\end{enumerate}


\subsection{Lebesgue Outer Measures}

\textbf{Countably Additivity}: $(E_n)_{n\in\mathbb{N}}$ disjoint sets, $m$ be measure. \ \ $\Rightarrow$ \ \ Countably Additive if: $m(\cup_{n}E_n)=\sum m(E_n)$

\textbf{Lebesgue Outer Measure}: $m^*(E):=\inf\left\{\sum_{n=1}^{\infty}\lambda(I_n) \ | \ E\subseteq\cup_{n=1}^{\infty}I_n,I_n \ is \ a \ open \ interval \right\}$ \ \ \ \ \ \ \ \ \ {\tiny ps: 本质是寻找到最小的\textit{区间}可以覆盖$E$,用前面$\lambda$的定义来定义"长度"}

$\cdot$ \textbf{Remark}: If it's impossible to find $(I_n)_{n\in\mathbb{N}}$ s.t. $\sum\lambda(I_n)<\infty$ \ \ $\Rightarrow$ \ \ $m^*(E)=\infty$ \ \ \ \ \ \ || \ \ \ \ \ \ Can interpret $m^*$ as $m^*:\mathcal{P}(\mathbb{R})\to[0,\infty)$

\begin{enumerate}[itemsep=-2pt, topsep=-2pt]
    \item \textbf{Lebesgue Outer Measure (Open)}: $m^*(E)=\inf\left\{\lambda(U),E\subseteq U,U \ is \ open \ set \right\}$ \ \ \ \ \ \ ps:$\lambda(U)=\sum \lambda(I_n)$
    \item \textbf{Lebesgue Outer Measure (Interval)}: If $J\subseteq\mathbb{R}$ is interval. $m^*(J)=\lambda(J)$
    \item \textbf{Useful Tool|By def}: $\forall E,\exists$ open interval $(I_n)_{n\in\mathbb{N}}$ s.t. $E\subseteq \cup I_n$ and $\sum\lambda(I_n)<m^*(E)+\varepsilon$
    \item \textbf{Useful Tool|By def}: $\forall E,\exists$ open set $U$ s.t. $E\subseteq U$ and $m^*(U)<m^*(E)+\varepsilon$
    \item \textbf{Useful Tool|Plus|By def}: $\forall E,\exists$ open intervals $(I_n)_{n\in\mathbb{N}}$, satisfy \ \ $^1${\footnotesize 第3条} \ \ $^2I_n\cap E\ne\emptyset$ \ \ $^3\lambda(I_n)<\delta$ \ \ $^4$
    \item \textbf{Futher|$m^{**}(E)$}: $m^*(E)=m^{**}(E):=\inf\left\{\sum\lambda(I_n) \ | \ E\subseteq \cup I_n,I_n \ interval \right\}$ \ \ \ \ \ \ || \ \ \ \ \ \ $m^*(E)\leq\sum\lambda(I_n),$ if $E=\cup I_n$
\end{enumerate}

\textbf{Properties of Lebesgue Outer Measure}: Let Set $E,F \ s.t. \ E\subseteq F\subset\mathbb{R}$. \ \ \ \ \ \ Emptyset: $\emptyset$

\begin{enumerate}[itemsep=-2pt, topsep=-2pt]
    \item \textbf{Null Empty Set}: $m^*(\emptyset)=0$.
    \item \textbf{Monotonicity}: $m^*(E)\leq m^*(F)$.
    \item \textbf{Countable Subadditivity}: $m^*(\cup_{n=1}^{\infty}E_n)\leq\sum_{n=1}^{\infty}m^*(E_n)$.
    \item Countable set $E$ \ \ $\Leftrightarrow$ \ \ $m^*(E)=0$ \ \ \ {\footnotesize Hint: $E=\cup^\infty_{j=1}\{x_j\}$} \ \ \ \ \ \ \ \ \ || \ \ \ \ \ \ \ \ \ $\exists m^*(E)=0,E$ uncountable {\footnotesize (e.g.Cantor Set)}
\end{enumerate}


\subsection{Countable Additive for Lebesgue Outer Measure}

\textbf{Def of $dist(E,F)$}: $dist(E,F):=\inf\{|x-y|:x\in E,y\in F\}$

\begin{enumerate}[itemsep=-2pt, topsep=-2pt]
    \item \textbf{Basic}: If $dist(E,F)>0$ \ \ $\Rightarrow$ \ \ $m^*(E\cup F)=m^*(E)+m^*(F)$.
    
    \vspace{-5pt}
    $\cdot$ $^1E,F$ are nonempty \textit{closed} sets; \ \ $^2E\cap F=\emptyset$; \ \ $^3E$ \textit{bounded}. \ \ $\Rightarrow$ \ \ $dist(E,F)>0$

    \item \textbf{Closed|Open|Interval Sets}: If $(K_n)_{n\in\mathbb{N}}$ is \textit{disjoint}, (Closed|Open|Interval), \textit{bounded} sets \ \ $\Rightarrow$ \ \ $m^*(\cup^\infty_{n}K_n)=\sum^\infty_{n=1} m^*(K_n)$
\end{enumerate}


\subsection{Lebesgue Measurable Sets}

\textbf{Lebesgue Measurable}: 

\begin{enumerate}[itemsep=-2pt, topsep=-2pt]
    \item \textbf{Version 1}: A set $E\subseteq\mathbb{R}$ is \textit{Lebesgue measurable} if $\forall\varepsilon>0,\exists$ open set $U\subseteq\mathbb{R}$ s.t. $E\subseteq U$ and $m^*(U\setminus E)<\varepsilon$ \ \ \ {\tiny ps: 被open set 从外"恰好"包住}
    \item \textbf{Version 2}: A set $E\subseteq\mathbb{R}$ is \textit{Lebesgue measurable} if $\forall\varepsilon>0,\exists$ closed set $U\subseteq\mathbb{R}$ s.t. $U\subseteq E$ and $m^*(E\setminus U)<\varepsilon$ \ \ \ {\tiny ps: 被closed set 从内"恰好"包住}
\end{enumerate}

$\cdot$ $\mathcal{M}:=$ collection of all Lebesgue measurable sets of $\mathbb{R}$.

\textbf{Property of Lebesgue Measurable Sets}:

\begin{enumerate}[itemsep=-2pt, topsep=-2pt]
    \item \textbf{Closed Set|Open Set|Interval}: Any closed set / open set / interval is Lebesgue measurable.
    \item \textbf{Union $\cup$|Intersection $\cap$}: if $(E_n)_{n\in\mathbb{N}}$ is \textit{Lebesgue Measurable} \ \ $\Rightarrow$ \ \ $^1E:=\cup^\infty_{n=1}E_n$ \ \ $^2E:=\cap^\infty_{n=1}E_n$ is \textit{Lebesgue Measurable}.
    \item \textbf{Complement}: if $E\subseteq\mathbb{R}$ is \textit{Lebesgue Measurable} \ \ $\Rightarrow$ \ \ $\mathbb{R}\setminus E$ is also \textit{Lebesgue Measurable}.
    \item \textbf{Measure Zero Set}: $m^*(E)=0$ \ \ $\Leftrightarrow$ \ \ $E$ Lebesgue measurable.
    \item \textbf{Translation Invariance}: $m^*(E)=m^*(x+E)$ \ \ \ \ \ \ || \ \ \ \ \ \ If $E$ is Lebesgue Measurable \ \ $\Rightarrow$ \ \ $x+E$ also.
\end{enumerate}


\subsection{Countable Additive \& Monotone Convergence for Lebesgue Measure}

\textbf{Lebesgue Measure}: \textit{Lebesgue Measure} is a function $m:\mathcal{M}\to[0,\infty)$ s.t. $m(E):=m^*(E),\forall m\in\mathcal{M}$ \ \ \ \ \ \ {\scriptsize ps: $\mathcal{M}$ 是所有Lebesgue可测集合的集合}

$\cdot$ \textbf{Property}: \textit{$\star$ All properties of Lebesgue Outer Measure hold for Lebesgue Measure. $\star$}  \ \ \ \ \ \ \ \ \ \ \ \ \ \ {\scriptsize ps: 原来是 $m^*:\mathcal{P}(\mathbb{R})\to[0,\infty)$,现在是 $m:\mathcal{M}\to[0,\infty)$}

\textbf{Countable Additivity for Measure}: If $(E_n)_{n\in\mathbb{N}}$ \textit{disjoint} \& \textit{Lebesuge Measurable}.\ \ $\Rightarrow$ \ \ $m(\cup^\infty_{n=1}E_n)=\sum^\infty_{n=1}m(E_n)$

\textbf{Monotone Convergence for Measure}: If $(E_n)_{n\in\mathbb{N}}$ \textit{Lebesuge Measurable}.

\begin{enumerate}[itemsep=-2pt, topsep=-2pt]
    \item \textbf{Increasing$\uparrow$}: If $E_n\subseteq E_{n+1}$ \ \ $\Rightarrow$ \ \ $m(\cup^\infty_{n=1}E_n)=\lim\limits_{n\to\infty} m(E_n)$
    \item \textbf{decreasing$\downarrow$}: If $E_{n+1}\subseteq E_n,m(E_1)<\infty$ \ \ $\Rightarrow$ \ \ $m(\cap^\infty_{n=1}E_n)=\lim\limits_{n\to\infty} m(E_n)$ \ \ \ {\footnotesize (Hint: $F_n:=E_n\setminus E,E=\cap E_n$)}
\end{enumerate}
