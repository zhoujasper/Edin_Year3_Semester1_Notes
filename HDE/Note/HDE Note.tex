\subsection{Linear Algebra || Calculus || SVCDE Method for ODE}

\textbf{Matrix Calculate}: $det(kA)=k^ndet(A)$ \ \ || \ \ $det(A^{-1})=det(A)^{-1}$ \ \ || \ \ $(AB)^*=B^*A^*$ \ \ || \ \ $det(A)=det(A^T)$.

$\cdot$ \textbf{Eigenvalue $\lambda$ \& Eigenvector $\mathbf{x}$}: Def: $A\mathbf{x}=\lambda\mathbf{x}$. \ \ || \ \ Method: $^1det(A-\lambda I)=0\to\lambda$. \ \ $^2(A-\lambda I)\mathbf{x}=0\to\mathbf{x}$.

$\cdot$ \textbf{Arithmetic Manipulate (AM) $\lambda$}: $\lambda$ 重复出现的次数. \ \ $\geq$ \ \ \textbf{Geometric Manipulate (GM) of $\lambda$}: \# $\mathbf{x}$ under $\lambda$.
    
$\cdot$ \textbf{Generalised Eigenvectors}: If AM>GM: Using $\xi_\lambda = (A-\lambda\mathbb{I})\eta_1,\eta_{n-1} = (A-\lambda\mathbb{I})\eta_{n}$ for each $\xi_\lambda$. \ \ \ \ \ \ $\eta_j$ are \textit{generalised eigenvectors}.

\textbf{Some Defs}: $\nabla F=(\partial_xF,\partial_yF,\partial_zF,...)$ \ \ \ \ \ \ || \ \ \ \ \ \ \textbf{Simply Connected}: { 没有洞,相互连接的区域}. \ \ \ \ \ \ || \ \ \ \ \ \ \textbf{$\partial D$}: { 区域 D 的边界.}

\textbf{Method of Integrating Factors}: $y'+p(x)y=q(x)$. \ \ $\Rightarrow$ \ \ Integrating Factor:$\mu=e^{\int p(x)dx}$ \ , \ $\mu \ y=\int q(x)\cdot \mu \ dx$

\textbf{Homogeneous 2nd ODE form}: $ay''+by'+cy=0$ \ \ $\Rightarrow$ \ \ Characteristic $ar^2+br+c=0$, \ \ \ \ { Roots: $r_1,r_2$} \ \ {\scriptsize ps:$y''\pm\lambda y=0$见附录}

\subsection{Linear || (ODE) Uniqueness Theorem || Autonomous}

\textbf{Def of Linear}: Let $L$ { 去掉 $y^{(n)}_i$ 的函数} {\scriptsize (e.g. $L=\frac{d^2}{dx^2}-x$ for $y''-xy=x^2$)} 

\hspace{28pt} $\Rightarrow$ \ \ Linear if: $L[a_1y_1+a_2y_2]=a_1L[y_1]+a_2L[y_2]$

\textbf{Existence \& Uniqueness Theorem (Real func)}:

$\cdot$ $^1\mathbf{y}'(t)=\mathbf{F}(t,\mathbf{y}^T)$, $\mathbf{y}(t_0)=\mathbf{y_0}$. \ \ \ \ \ \ \ { Let $R_i=[t_0-\delta,t_0+\delta]\times\prod_j[y_{0_j}-\delta_j,y_{0_j}+\delta_j]$.}

$\cdot$ $^2F_j  \ , \ \partial_{y_i}F_j:R_i \to \mathbb{R}$ are continuous.  \ \ $\Rightarrow$ \ \ $\exists$ \underline{unique} solution $\mathbf{y}$ in $R'_i$

\hspace{85pt} $R'_i=[t_0-\varepsilon,t_0+\varepsilon]\times\prod_j[y_{0_j}-\varepsilon_j,y_{0_j}+\varepsilon_j]\subseteq R_i$.

\textbf{Autonomous}: For $\mathbf{y}'(t)=\mathbf{f}(t,\mathbf{y}^T)$. \ \ $\partial_x\mathbf{f}=\mathbf{0}$ $\Rightarrow$ autonomous. \ \ \ { or: $\mathbf{f}(t,\mathbf{y}^T)=\mathbf{f}(\mathbf{y}^T)$}


\section{Linear ODEs}

\textbf{n-th ODE $\rightarrow$ 1st ODEs}: For  $y^{(n)}=F(t,y,y',...,y^{(n-1)})$. \ \ \ $\Rightarrow$ \ \ \ Let $x_i=y^{(i-1)}$.

\textbf{Def of Linear 1st ODEs}: $\mathbf{y}'(t) = P(t)\mathbf{y}(t) + \mathbf{g}(t)$ (\star), {\tiny $P$ is matrix-valued function.}

\hspace{137pt} \textbf{Homogeneous}: $\mathbf{g}(t)=\mathbf{0}$.

\textbf{Existence \& Uniqueness Theorem}: If $P_{ij}$, $\mathbf{g}$ of (\star) continuous in $R$, $\mathbf{y}(t_0)=\mathbf{y}_0$

\ \ $\Rightarrow$ \ \ $\exists$ \underline{unique} solution $\mathbf{y}$ in $R'$ \ \ \ {\tiny $R'=[t_0-\varepsilon,t_0+\varepsilon]\subseteq R=[t_0-\delta,t_0+\delta]$}

For $\lambda,\overline{\lambda}\in\mathbb{C}$. \ \ $\Rightarrow$ \ \ $e^{\lambda t}\xi=\mathbf{u}+\mathbf{v}i$. \ \ $\Rightarrow$ \ \ Independent sol: $\mathbf{u},\mathbf{v}$ \ \ $\Rightarrow$ \ \ General solution: $\mathbf{y} = c_1\mathbf{u} + c_2\mathbf{v} + \cdots$

    \textbf{Matrix Method}: $\mathbf{y}(t)=e^{At}\mathbf{y}(0)$ \ \ \ \ \ \ \ \ \ \ \ { (分两种方法: 1.直接展开; 2.通过一些已有 property 计算. ps: $e^{At}=\Psi$)}

    $\cdot$ \textbf{Property|AM>GM}: \ $e^{At}=Te^{Jt}T^{-1}$ \ , \ $T=[\lambda_1,...,\lambda_i,\eta^{(i)}_1,...,\lambda_j,\eta^{(j)}_1,...]$ \ , \ $J=T^{-1}AT$ { (Jordan form)} \ , \ $e^{J_it}=$ 见右边

    \textbf{Diagonalisation Method}: (Ctd.) Let $\mathbf{y}=T\mathbf{x}$ \ \ $\Rightarrow \ \ \mathbf{x}'=J\mathbf{x}$ [{ or} $\mathbf{x}'=D\mathbf{x}$] { 按前面方法解}

    $\cdot$ \textbf{Relation with $e^{At}$}: $e^{At}=\Psi(t)$ with $\Psi(0)=\mathbb{I}$ \ \ \ || \ \ \ $e^{At}=\Psi(t)\Psi^{-1}(0)$

    \textbf{Stable}: Critical Point $\mathbf{x}_0$ is stable \ \ iff \ \ $\forall\varepsilon>0,\exists\delta>0$ s.t. $\forall$ sols $\mathbf{x}(t)$ with $||\mathbf{x}(0)-\mathbf{x}_0||<\delta$, then $||\mathbf{x}(t)-\mathbf{x}_0||<\varepsilon$,$\forall t\geq0$ \ \ || \ \ 直观理解: 开始近 $\to$ 保持近.

    \textbf{Attracting}: Critical point $\mathbf{x}_0$ is attracting \ \ iff \ \ $\exists\delta>0$ s.t. $\forall$ sols $\mathbf{x}(t)$ with $||\mathbf{x}(0)-\mathbf{x}_0||<\delta$ satisfies: $\lim_{t\to\infty}\mathbf{x}(t)=\mathbf{x}_0$ \ \ || \ \ 直观理解: 开始近 $\to$ 不断趋向.

    \textbf{Asymptotically Stable}: Critical point $\mathbf{x}_0$ is asy.stab \ \ iff \ \ $^1$ Stable, $^2$ Attracting.

    \textbf{Theorem}: If $\mathbf{x}_\ast(t)=(x_\ast,y_\ast)$ s.t. $F(\mathbf{x}_\ast),G(\mathbf{x}_\ast)\ne0$. \ \ $\Rightarrow$ \ \ $\exists H$ s.t. $H(x,y)=(\tilde{x},\tilde{y}),\tilde{x}'=1,\tilde{y}'=0$ {\tiny H is a smooth change of variable function in neighbourhood of $\mathbf{x}_\ast$}

    \textbf{Non-Linear $\to$ Linear}: For system (\ast\ast), assume $\mathbf{x}_0=(x_0,y_0)$ is critical point. \ $\Rightarrow$ \ $\mathbf{u}=\mathbf{x}-\mathbf{x}_0$, then $\mathbf{u}'\approx A\mathbf{u}${\tiny \ Linear;} {\tiny $A=\begin{pmatrix}\partial_xF(\mathbf{x}_0) & \partial_yF(\mathbf{x}_0) \\ \partial_xG(\mathbf{x}_0) & \partial_yG(\mathbf{x}_0) \\ \end{pmatrix}$} {\tiny Jacobian}

\subsection{Fourier Series}

\textbf{Fourier Series}: Periodic function with $T=2L$ \ , \ $f(x)=\frac{a_0}{2}+\sum_{n=1}^{\infty}\left(a_ncos(\frac{n\pi x}{L})+b_nsin(\frac{n\pi x}{L})\right)$ (*)

$\cdot$ \textbf{Euler-Fourier Formulas}: Suppose (*) converges in the interval $[-L,L]$. Then: $a_n=\frac{1}{L}\int^L_{-L}f(x)cos(\frac{n\pi x}{L})dx$ \ \ \ \ \ \ \ \ $b_n=\frac{1}{L}\int^L_{-L}f(x)sin(\frac{n\pi x}{L})dx$ \ \ \ \ \ \ \ \ $a_0=\frac{1}{L}\int^L_{-L}f(x)dx$ {\scriptsize (Average Value of $f$ over $2L$)}

\vspace{2pt}
\textbf{Piecewise Continuous}: { $f(x)$ piecewise continuous on $[a,b]$ \ \ iff \ \ $\exists a=x_0<\cdots<x_n=b$ s.t. \ \ \ $^1f(x)$ continuous $\forall x\in(x_{i-1},x_i)$ \ \ \ $^2 f(x_i^{\pm})=\lim_{x\to x_i^{\pm}}f(x)$ finite.}

\textbf{Term-By-Term Integration}: Fourier Series for \textit{piecewise continuous} functions can be \underline{Integrated term-by-term} to \underline{Convergent} Fourier Series.

\textbf{Finite Truncation (Partial Sum)}: { Finite Truncation of Fourier Series is: $s_k=\frac{a_0}{2}+\sum_{n=1}^{k}\left(a_ncos(\frac{n\pi x}{L})+b_nsin(\frac{n\pi x}{L})\right)$ \ \ || \ \  \textbf{Error}: $e_N=s_N-f(x)$}

\subsection{Sturm-Liouville Theory}

\textbf{Sturm-Liouville (SL) Problem}: Form: $-(p(x)y')'+q(x)y=\lambda r(x)y$ \ , \ $p,r>0$ \ \ || \ \ With: $a_1y(a)+a_2y'(a)=b_1y(b)+b_2y'(b)=0$ \ \ $p,p',q,r$ continuous. \ \ || \textbf{weight function}: $r(x)$. \ \ \ || \ \ \ {\tiny $\Downarrow$ \textbf{Lagrange's Identity}}

$\cdot$ $\Rightarrow$ Let \textit{Linear Operator}: $\mathcal{L}(y):=-(p(x)y')'+q(x)y$ \ \ \ $\Rightarrow$ \ \ \ Problem becomes: $\mathcal{L}(y)=\lambda r(x)y$ \ \ {\tiny (It's a Eigenvalue Problem $\mathcal{L}(\Psi_n)=\lambda_n r(x)\Psi_n$)} \ \ $\Rightarrow$ \ \ $\forall$ solutions $u,v$ with (二阶导连续) of SL Problem: $(\mathcal{L}(u),v)=(u,\mathcal{L}(v))$ !!!

$\cdot$ \textbf{Properties|Real|Orthogonal}: In SL Problem: $^1$ $\lambda_n$, $\Psi_n$ are \textit{real}; \ \ \ \ \ || \ \ \ \ \ $^2$ Eigenfunctions $\Psi_1,\Psi_2$ with $\lambda_1\ne\lambda_2$. \textbf{Modified Inner Product}: $\langle \Psi_1,\Psi_2 \rangle=\int^b_a r(x)\Psi_1\Psi_2 dx=0$ \ \ {\tiny $i.e.\Psi_1,\Psi_2$ orthogonal.}

$\cdot$ \textbf{Properties|Simple|Ordered}: In SL Problem: $^3$ $\lambda_n$ are \textit{simple}: \ \ $\forall\lambda_n$ has \textit{unique} $\Psi_n$; \ \ \ \ \ || \ \ \ \ \ $^4\lambda_n$ are \textit{Ordered}: $\lambda_1<\lambda_2<\cdots<\lambda_n<\cdots\to\infty$ as $n\to\infty$

\textbf{Normalisation of Eigenfunction}: $^1\Psi_n$ normalised if: $\langle \Psi_n,\Psi_n \rangle=1$ \ \ \ || \ \ \ $^2$ Method (Normalised $\Psi_n$): $\Phi_n=\frac{1}{\sqrt{\langle \Psi_n,\Psi_n \rangle}}\Psi_n=[\int^b_ar(x)\Psi^2_ndx]^{-1/2}\Psi_n$

\textbf{Sturm-Liouville Convergence Theorem}: If $f,f'$ are piecewise continuous on $[0,1]$ \ \ $\Rightarrow$ \ \ $F(x)=\sum_{n=1}^{\infty}c_n\Phi_n$ converges. \ \ \ || \ \ \ $\Phi_n$ are \textit{orthonormal} eigenfunction of a Sturm-Liouville problem.

$\cdot$ Futhermore, $^1F\to f$ when $f$ is continuous; \ \ \ \ \ \ $^2F\to\frac{f(x^+)+f(x^-)}{2}$ when $f$ is discontinuous. \ \ {\tiny 边界点右边界条件决定} \ \ \ \ \ \ \textbf{Calculate Coefficients}: $c_n=\langle f,\Phi_n\rangle=\int^b_ar(x)f(x)\Phi_n(x)dx$ \ \ \ \ || \ \ \ \ {\tiny ps: $c_n$ minimize the MSE $\varepsilon_N$}

$\cdot$ 如果我们不需要归一化: 特征函数 $\Psi_n$ \ \ $\Rightarrow$ \ \ $f(x)=\sum^\infty_{n=1}c_n\Psi_n$ , $c_n=\langle f,\Psi_n\rangle=\int^b_af(x)\Psi_n(x)dx$ \ \ \ \ \ || \ \ \ \ \ \textbf{Parseval's Identity}: $f,f'$ are piecewise continuous on $[a,b]$ \ \ $\Rightarrow$ \ \ $\langle f,f\rangle=\int^b_af^2(x)r(x)dx=\sum^{\infty}_{n=1}c^2_n$

\textbf{Hyperbolic Func}: $sinh(x)=\frac{e^x-e^{-x}}{2}$ \ \ \ $cosh(x)=\frac{e^x+e^{-x}}{2}$ \ \ \ $cosh^2(x)-sinh^2(x)=1$ \ \ \ $[sinh(x)]'=cosh(x)$ \ \ \ $[cosh(x)]'=sinh(x)$ \ \ || \ \ $\sinh(x \pm y) = \sinh(x)\cosh(y) \pm \cosh(x)\sinh(y)$ \ \ \ \ \ \ $\cosh(x \pm y) = \cosh(x)\cosh(y) \pm \sinh(x)\sinh(y)$
