\section{Basic Knowledge \& SVCDE Methods for ODE}

\subsection{Linear Algebra}

\textbf{Matrix Calculate}: $det(kA)=k^ndet(A)$ \ \ || \ \ $det(A^{-1})=det(A)^{-1}$ \ \ || \ \ $(AB)^*=B^*A^*$ \ \ || \ \ $det(A)=det(A^T)$.

\textbf{Eigenvalue $\lambda$ \& Eigenvector $\mathbf{x}$}: Def: $A\mathbf{x}=\lambda\mathbf{x}$. \ \ \ \ \ \ \ \ \ \ \ \ Method: $^1det(A-\lambda I)=0\to\lambda$. \ \ \ $^2(A-\lambda I)\mathbf{x}=0\to\mathbf{x}$.

$\cdot$ {\footnotesize \textbf{Algebraic Multiplicity (AM) of $\lambda$}: $\lambda$ 重复出现的次数. \ \ \ \ \ \ \ $\geq$ \ \ \ \ \ \ \ \textbf{Geometric Multiplicity (GM) of $\lambda$}: \# $\mathbf{x}$ under $\lambda$.}

$\cdot$ {\footnotesize \textbf{Generalised Eigenvectors}: If AM>GM: Using $\xi_\lambda = (A-\lambda\mathbb{I})\eta_1,\eta_{n-1} = (A-\lambda\mathbb{I})\eta_{n}$ for each $\xi_\lambda$. \ \ \ \ \ \ $\eta_j$ are \textit{generalised eigenvectors}.}


\subsection{Calculus \& SVCDE Method for ODE}

\textbf{Some Defs}: $\nabla F=(\partial_xF,\partial_yF,\partial_zF,...)$ \ \ \ \ \ \ || \ \ \ \ \ \ \textbf{Simply Connected}: {\footnotesize 没有洞,相互连接的区域}. \ \ \ \ \ \ || \ \ \ \ \ \ \textbf{$\partial D$}: {\footnotesize 区域 D 的边界.}

\textbf{Method of Integrating Factors}: $y'+p(x)y=q(x)$. \ \ $\Rightarrow$ \ \ Integrating Factor:$\mu=e^{\int p(x)dx}$ \ , \ $\mu \ y=\int q(x)\cdot \mu \ dx$

\textbf{Homogeneous 2nd ODE form}: $ay''+by'+c=0$ \ \ $\Rightarrow$ \ \ Calculate $ar^2+br+c=0$, \ \ \ \ {\footnotesize ps: $\Delta=b^2-4ac$, roots: $r_1,r_2$}

$\cdot$ {\footnotesize 1. $\Delta>0$,$y=C_1e^{r_1x}+C_2e^{r_2x}$; \ \ \ \ \ \ \ \ 2. $\Delta=0$,$y=(C_1+C_2x)e^{rx}$; \ \ \ \ \ \ \ \ 3. $\Delta<0$,$y=e^{\alpha x}[C_1cos(\beta x)+C_2sin(\beta x)]$ \ \ ps: $r_1,r_2 = \alpha\pm\beta i$}

\textbf{non-Homogeneous 2nd ODE form}: $ay''+by'+c=g(x)$ \ \ $\Rightarrow$ \ \ $y=y_h+y_p$ \ \ \ {\footnotesize ps:$y_h$ is sol for $g(x)=0$.}

$\cdot$ {\footnotesize Method of Undetermined Coefficients: Let $A_n(x)=\sum_{i=0}^{n}a_ix^i, \ \ B_n(x)=\sum_{i=0}^{n}b_ix^i, \ \ P_n(x)=\sum_{i=0}^{n}p_ix^i$.}

{
\footnotesize
\begin{enumerate}[itemsep=-2pt, topsep=-2pt]
   \item If $g(x)=P_n(x)$ \ $\Rightarrow$ \ $y_p=x^sA_n(x)$ \ , $s:= \ \ 0,(0\ne r_1,r_2); \ \ 1,(0 = r_1 \ or \ r_2); \ \ 2,(0 = r_1 = r_2).$
   \item If $g(x)=Me^{kx}$ \ $\Rightarrow$ \ $y_p=Cx^se^{kx}$ \ , $s:= \ \ 0,(k\ne r_1,r_2); \ \ 1,(k = r_1 \ or \ r_2); \ \ 2,(k = r_1 = r_2).$
   \item If $g(x)=Mcos(kx)+Nsin(kx)$ \ $\Rightarrow$ \ $y_p=x^s(C_1cos(kx)+C_2sin(kx))$ \ , $s:= \ \ 0,(r_1,r_2 \ne ik); \ \ 1,(r_1,r_2= ik).$
   \item If $g(x)=$ product from 1,2,3. \ $\Rightarrow$ \ $y_p=$ corresponding product of $y_p$ ($x^s$ 只乘一次) \ , \ $s:=$ 令$y_p$与$y_c$没有相同项. ($\leq 2$)
\end{enumerate}
}

\textbf{Euler Equation}: General: $a_nx^ny^{(n)}+a_{n-1}x^{n-1}y^{(n-1)}+\cdots+a_0y=0$ \ \ $\Rightarrow$ Seek Solution as: $y=x^\alpha$ \ \ \ \ $\Rightarrow$ solution: $y=\sum_ic_ix^{\alpha_i}$


\subsection{ODE / PDE Definitions}

\textbf{Def of DEs / ODEs / PDEs}: Given independent variables {\small ( $x_1,...,x_n$ )}, functions {\small ( $y_1(x_{j_1}),...,y_p(x_{j_p})$ )} and some derivatives {\small (e.g. $\partial_s y_t$).}

$\cdot$ \textbf{DEs}: $\mathcal{F}_i[y_r,...,\partial_{x_t}y_b,x_1,...,x_n]=0,\ \ \  i=s,s-1,...,1.$ \ \ \ \ \ \ \ || \ \ \ \ \ \ \ \textbf{ODEs}: $n=1$ {\footnotesize (无偏导)} \ \ \ \ \ \ \ \textbf{PDEs}: $n>1$ {\footnotesize (有偏导)}

\textbf{Def of Linear}: Let $L$ {\footnotesize 去掉 $y^{(n)}_i$ 的函数} {\scriptsize (e.g. $L=\frac{d^2}{dx^2}-x$ for $y''-xy=x^2$)} \ \ \ \ $\Rightarrow$ \ \ \ \ Linear if: $L[a_1y_1+a_2y_2]=a_1L[y_1]+a_2L[y_2]$


\subsection{(ODE) Uniqueness Theorem \& Autonomous}

\textbf{Existence \& Uniqueness Theorem (Real func)}: Given $\mathbf{y}'(t)=\mathbf{F}(t,\mathbf{y}^T)$, with $\mathbf{y}(t_0)=\mathbf{y_0}$. \ \ \ \ \ \ \ {\footnotesize Let $R_i=[t_0-\delta,t_0+\delta]\times\prod_j[y_{0_j}-\delta_j,y_{0_j}+\delta_j]$.}

$\cdot$ If $F_j  \ , \ \partial_{y_i}F_j:R_i \to \mathbb{R}$ are continuous.  \ \ $\Rightarrow$ \ \ $\exists$ \underline{unique} solution $\mathbf{y}$ in $R'_i$ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ {\footnotesize Let $R'_i=[t_0-\varepsilon,t_0+\varepsilon]\times\prod_j[y_{0_j}-\varepsilon_j,y_{0_j}+\varepsilon_j]\subseteq R_i$.}

\textbf{Autonomous}: For $\mathbf{y}'(t)=\mathbf{f}(t,\mathbf{y}^T)$. \ \ \ \ \ \ If $\partial_x\mathbf{f}=\mathbf{0}$ $\Rightarrow$ autonomous. \ \ \ {\footnotesize or: $\mathbf{f}(t,\mathbf{y}^T)=\mathbf{f}(\mathbf{y}^T)$ {\tiny (不含$t$)}}


\section{Linear ODEs}

\textbf{n-th ODE $\rightarrow$ 1st ODEs}: For {\small $y^{(n)}=F(t,y,y',...,y^{(n-1)})$}. \ \ \ $\Rightarrow$ \ \ \ Let $x_i=y^{(i-1)}$. Then $x'_n=F(t,x_1,x_2,...,x_{n-1})$ {\footnotesize which is 1st ODE.}

\textbf{Def of Linear 1st ODEs}: $\mathbf{y}'(t) = P(t)\mathbf{y}(t) + \mathbf{g}(t)$ (\star), {\tiny $P$ is matrix-valued function.} \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \textbf{Homogeneous}: $\mathbf{g}(t)=\mathbf{0}$.

\textbf{Existence \& Uniqueness Theorem}: If $P_{ij}$, $\mathbf{g}$ of (\star) continuous in $R$, $\mathbf{y}(t_0)=\mathbf{y}_0$ \ \ $\Rightarrow$ \ \ $\exists$ \underline{unique} solution $\mathbf{y}$ in $R'$ \ \ \ {\tiny $R'=[t_0-\varepsilon,t_0+\varepsilon]\subseteq R=[t_0-\delta,t_0+\delta]$}


\subsection{Homogeneous}

\vspace{-3pt}
\begin{center}
    Solve: $\mathbf{y}'=P\mathbf{y}$ \ \ \ ||\ \ \ If $P$ constant, $P=A$ \ \ \ || \ \ \ If possible: $\lambda,\xi,\eta$ be eigenvalues, eigenvectors, generalised eigenvectors.
\end{center}
\vspace{-10pt}

\textbf{Principle of Superposition}: If $\mathbf{y}_i$ are solutions $\Rightarrow \ \sum_ic_i\mathbf{y}_i$ also. \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ {\footnotesize $S:=\{\mathbf{y}|\mathbf{y}'=P\mathbf{y}\}$ is a \textit{linear space}, $dim(S)=n$ if $P_{ij}$ continuous.}

\textbf{Wronskian at $x$}: $W[\mathbf{y}^{(1)},...,\mathbf{y}^{(n)}](t)=$ {\tiny $\begin{vmatrix}y^{(1)}_1(t) & \cdots & y^{(n)}_1(t) \\ \cdots & \cdots & \cdots\\ y^{(1)}_n(t) & \cdots &y^{(n)}_n(t) \end{vmatrix}$}. \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ $W(t)\ne0$,$\forall t$ \ $\Rightarrow \ \{\mathbf{y}^{(j)}\}$ \textbf{linearly independent}. {\footnotesize ps:经常用 Abel Theorem}

$\cdot$ Geoemtric Meaning: {\footnotesize Geometrically, $W$ is the volume spanned by the three vectors $y^{(1)}, y^{(2)}, y^{(3)}$; \ \ $t$ 增加或减小代表在 $y^{(i)}$方向上 compress/stretch.}

\textbf{Abel's Theorem}: If $W(t_0)\ne0$,$t_0\in I$ \ \ $\Rightarrow$ \ \ $\forall t\in I,W(t)\ne0$ \ \ || \ \ \textbf{Abel's Formula}: $W(t)=W(t_0)e^{\int^t_{t_0}tr[P(s)] ds};t_0,t\in I$ {\tiny $P(t)$ is continuous in $I$.}

\textbf{Fundamental Set of Solutions (FSoS)}: If $\{\mathbf{y}^{(j)}(t)\}$ \textit{linearly independent} at $t\in I$, it is FSoS in $I$.

\textbf{General Solution}: If $\{\mathbf{y}^{(j)}(t)\}_{j=1}^n$ is FSoS (Linearly Independent) \ \ $\Rightarrow$ \ \ General solution: $\mathbf{y}=\sum_{j=1}^nc_j\mathbf{y}_j$


\subsubsection{Eigenvalue/vector Method ($A$ constant)}

\textbf{AM=GM|Real $\lambda$}: general solution: $\mathbf{y}=\sum_jc_je^{\lambda_jt}\xi_j$.

\textbf{AM=GM|Complex $\lambda$}: (ctd.) For $\lambda,\overline{\lambda}\in\mathbb{C}$. \ \ $\Rightarrow$ \ \ $e^{\lambda t}\xi=\mathbf{u}+\mathbf{v}i$. \ \ $\Rightarrow$ \ \ Independent sol: $\mathbf{u},\mathbf{v}$ \ \ $\Rightarrow$ \ \ General solution: $\mathbf{y} = c_1\mathbf{u} + c_2\mathbf{v} + \cdots$

\textbf{AM>GM}: (ctd.) {\scriptsize (For each $\xi_\lambda$, under $\lambda$)} {\small $\mathbf{y}^{(1)}=e^{\lambda t}\xi_\lambda$ \ \ $\mathbf{y}^{(2)}=e^{\lambda t}(t\xi_\lambda+\eta_1)$ \ \ $\mathbf{y}^{(3)}=e^{\lambda t}(\frac{t^2}{2!}\xi_\lambda+t\eta_2+\eta_1)$, ... ,$\mathbf{y}^{(n)}=e^{\lambda t}[\frac{t^{n-1}}{(n-1)!}\xi_\lambda+\sum_{i=0}^{n-2}\frac{t^{i}}{(i)!}\eta_{i+1}]$}


\subsubsection{Matrix Method | Exponential}

\textbf{Matrix Method}: $\mathbf{y}(t)=e^{At}\mathbf{y}(0)$ \ \ \ \ \ \ \ \ \ \ \ {\footnotesize (分两种方法: 1.直接展开; 2.通过一些已有 property 计算. ps: $e^{At}=\Psi$)}

$\cdot$ \textbf{Property|AM=GM}: \ $e^{At}=Te^{Dt}T^{-1}$ \ , \ $T=[\xi_1,...,\xi_n]$ \ , \ $D=diag(\lambda_1,...,\lambda_n)$ \ , \ $e^{Dt}=diag(e^{\lambda_1t},...,e^{\lambda_nt})$

\vspace{-20pt}
$\cdot$ \textbf{Property|AM>GM}: \ $e^{At}=Te^{Jt}T^{-1}$ \ , \ $T=[\lambda_1,...,\lambda_i,\eta^{(i)}_1,...,\lambda_j,\eta^{(j)}_1,...]$ \ , \ $J=T^{-1}AT$ {\footnotesize (Jordan form)} \ , \ $e^{J_it}=e^{\lambda_i t}$ {\footnotesize $\begin{pmatrix} 1 & t & \frac{t^2}{2} & \cdots & \frac{t^{n-1}}{(n-1)!} \\ 0 & 1 & t & \cdots & \frac{t^{n-2}}{(n-2)!} \\ \vdots & \vdots & \vdots & \ddots & \vdots \\ 0 & 0 & 0 & \cdots & 1 \end{pmatrix}$}
\vspace{-20pt}

\textbf{Diagonalisation Method}: (Ctd.) Let $\mathbf{y}=T\mathbf{x}$ \ \ $\Rightarrow \ \ \mathbf{x}'=J\mathbf{x}$ [{\footnotesize or} $\mathbf{x}'=D\mathbf{x}$] {\footnotesize 按前面方法解}

\textbf{Exponential of a matrix}: $e^{At}:=\sum_{n=0}^{\infty}\frac{(At)^n}{n!} \ = \ \mathbb{I}+At+\frac{1}{2!}A^2t^2+\cdots \ = \ \lim_{n\to\infty}(\mathbb{I}+\frac{At}{n})^n$

$\cdot$ \textbf{Property}: $e^{A+B}=e^{A}e^{B}$ if $AB=BA$ \ \ \ || \ \ \ $e^{A\cdot0}=\mathbb{I}$ \ \ \ || \ \ \ $[e^{At}]'=Ae^{At}$


\subsubsection{Matrix Method | Fundamental Matrix}

\textbf{Fundamental matrix $\Psi$}: If $\{\mathbf{y}^{(j)}(t)\}$ is FSoS. \ \ $\Rightarrow$ \ \ $\Psi(t)=[\mathbf{y}^{(1)}(t),...,\mathbf{y}^{(n)}(t)]$.

$\cdot$ \textbf{Property}: $\Psi'(t)=P(t)\Psi(t)$ (It's a sol). \ \ \ || \ \ \ $det[\Psi(t)]=W(t)\ne0$ \ \ \ || \ \ \ general solution: $\mathbf{y}=\Psi\mathbf{c}$ \ , \ $\mathbf{c}=\Psi^{-1}(t_0)\mathbf{y}_0$

$\cdot$ \textbf{Relation with $e^{At}$}: $e^{At}=\Psi(t)$ with $\Psi(0)=\mathbb{I}$ \ \ \ || \ \ \ $e^{At}=\Psi(t)\Psi^{-1}(0)$


\subsection{non-Homogeneous}

\textbf{Theorem}: Solution of (\star): $\mathbf{y}=\sum_ic_i\mathbf{y}^{(i)}+\mathbf{y}_{p}$, \ \ where $\mathbf{y}^{(i)}$ is solution of it's Homogeneous. \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ \ {\tiny $\Downarrow$ 通常会让特解中积分后的常数c=0}

\textbf{Variation of Parameters}: $\mathbf{y}=\mathbf{y}_g+\mathbf{y}_p$; \ \ \ $\mathbf{y}_g=\Psi(t)\Psi^{-1}(t_0)\mathbf{y}(t_0)$; \ \ \ $\mathbf{y}_p=\Psi(t)\mathbf{u}(t),\mathbf{u}(t)=\int^t_{t_0}\Psi^{-1}(s)\mathbf{g}(s)ds$. \ \ {\scriptsize 经常$\Psi(t)\mathbf{u}'=\mathbf{g}$来找$\mathbf{u}$}

\textbf{Matrix methods|Diagonalisation}: $^1$Let $\mathbf{y}=T\mathbf{x}$ \ \ \ $^2(\star)\Rightarrow \mathbf{x}'=D\mathbf{x}+T^{-1}\mathbf{g}=D\mathbf{x}+\mathbf{h}$ \ \ \ $^3$ {\tiny 直接计算;} $z_i=c_ie^{r_it}+\underline{e^{r_it}\int^t_{t_0}e^{-r_is}h_i(s)ds}_{\mathbf{y}_p}$

\textbf{Undetermined Coefficients}: {\footnotesize when $\mathbf{g}(t)$ is \underline{polynomials}, or \underline{real/complex exponential}.} \ \ \ Let {\footnotesize $P_n(t)=\sum^n_{i=0}\mathbf{p}_it^i,A_n(t)=\sum^n_{i=0}\mathbf{a}_it^i$}

$^1${\scriptsize 构造$\mathbf{y}_p$与$\mathbf{g}$ 结构相匹配.(e.g. If $\mathbf{g}=P_n(t)\Rightarrow \mathbf{y}_p=A_n(t)$; If $\mathbf{g}=\mathbf{c}e^{kt}\Rightarrow \mathbf{y}_p=\mathbf{a}e^{kt}$)} \ $^2$ {\scriptsize 特殊: If $\mathbf{g}=\mathbf{c}e^{\lambda t}$,$\lambda$ is eigenvalue with $AM=n\Rightarrow \mathbf{y}_p=e^{\lambda t}\sum_{i=0}^{n}\mathbf{a}_it^i$} \ $^3${\scriptsize 带入原方程得未知系数}


\subsection{Geometrical Interpretation \& Linear Approximation}

\vspace{-0.25cm}
{\tiny * 此节中，Consider nonlinear autonomous system: $x'=F(x,y),y'=G(x,y)$ (\ast\ast)}
\vspace{-0.15cm}

\textbf{Phase Plane}: {\small $O_{xy}$ referred as phase plane. \ \ \ \textbf{Trajectories}: Sol of ODE. $\mathbf{x}(t)=(x(t),y(t))^T$ {\tiny direction $(dx/dt,dy/dt)$} \ \ \ \textbf{Phases Portrait}:{\scriptsize 一组有代表性的轨迹}}

\textbf{Critical Point (Equilibrium Point)}: $\mathbf{x}_0=(x_0,y_0)$ is critical point iff $[x'=F(\mathbf{x}_0)=0,y'=G(\mathbf{x}_0)=0]$ {\scriptsize or} $[\mathbf{x}'=\mathbf{0}]$ {\tiny (critical point外变化简单)}

\textbf{Stable}: Critical Point $\mathbf{x}_0$ is stable \ \ iff \ \ $\forall\varepsilon>0,\exists\delta>0$ s.t. $\forall$ sols $\mathbf{x}(t)$ with $||\mathbf{x}(0)-\mathbf{x}_0||<\delta$, then $||\mathbf{x}(t)-\mathbf{x}_0||<\varepsilon$,$\forall t\geq0$

$\cdot$ {\footnotesize 直观理解:一开始离 critical point 很近,那么随着时间推移也会保持在 critical point 附近. \ \ \ ps:attrcting 是会趋近于 critical point.}

\textbf{Attracting}: Critical point $\mathbf{x}_0$ is attracting \ \ iff \ \ $\exists\delta>0$ s.t. $\forall$ sols $\mathbf{x}(t)$ with $||\mathbf{x}(0)-\mathbf{x}_0||<\delta$ satisfies: $\lim_{t\to\infty}\mathbf{x}(t)=\mathbf{x}_0$

\textbf{Asymptotically Stable}: Critical point $\mathbf{x}_0$ is asy.stab \ \ iff \ \ $^1$ Stable, $^2$ Attracting.

\textbf{Theorem}: If $\mathbf{x}_\ast(t)=(x_\ast,y_\ast)$ s.t. $F(\mathbf{x}_\ast),G(\mathbf{x}_\ast)\ne0$. \ \ $\Rightarrow$ \ \ $\exists H$ s.t. $H(x,y)=(\tilde{x},\tilde{y}),\tilde{x}'=1,\tilde{y}'=0$ {\tiny H is a smooth change of variable function in neighbourhood of $\mathbf{x}_\ast$}

\textbf{Non-Linear $\to$ Linear}: For system (\ast\ast), assume $\mathbf{x}_0=(x_0,y_0)$ is critical point. \ $\Rightarrow$ \ $\mathbf{u}=\mathbf{x}-\mathbf{x}_0$, then $\mathbf{u}'\approx A\mathbf{u}${\tiny \ Linear;} {\tiny $A=\begin{pmatrix}\partial_xF(\mathbf{x}_0) & \partial_yF(\mathbf{x}_0) \\ \partial_xG(\mathbf{x}_0) & \partial_yG(\mathbf{x}_0) \\ \end{pmatrix}$} {\tiny Jacobian}

\subsection{Function Space, Eigenfunctions, Periodic Function and Orthogonal Function}

\textbf{Periodic Functions}: Function $f(x)$ is Periodic iff $f(x+T)=f(x),\forall x$. \ \ \ \textbf{Period}: T. \ \ \ \textbf{Fundamental Period}: The smallest possible T.

$\cdot$ Property: If $f,g$ has same period $T$. \ $\Rightarrow$ \ $af(x)+bg(x)$ also has period $T$. \ \ \ || \ \ \ For $sin(\omega x),cos(\omega x)$: $T=2\pi/\omega$

\textbf{Differential Operators}: Any linear combination of $n$-th derivatives defines a linear map or linear operator. \ \ \ \ \ \ {\tiny e.g. $p(x)\frac{d}{dx}+\frac{d^2}{dx^2}:y\mapsto p(x)\frac{dy}{dx}+\frac{d^2y}{dx^2}$}

\textbf{Eigenvalues \& Eigenfunctions}: If $D$ is a \textit{differential operator} \ $\Rightarrow$ \ If $Df=\lambda f$, then $\lambda$ is \textbf{Eigenvalue} and $f$ is \textbf{Eigenfunctions}

$\cdot$ \textbf{Property}: {\footnotesize 对于微分方程: $Df=\lambda f$ \ \ $\Rightarrow$ \ \ solution: $f=\sum c_nf_n$ \ \ $f_n$ are Eigenfunctions}

\textbf{Inner Product}: $(f,g) =\int^b_{a}f(x)\overline{g(x)}dx$ \ \ \ \ \ \ For funcs $f,g$ in $(a,b)$ \ \ \ \ || \ \ \ \ \textbf{Orthogonal}: $(f,g)=0$ \ \ \ \ || \ \ \ \ \textbf{Square Norm}: $(f,f)$.

% $\cdot$ Speical Case: Let $C_n=cos(\frac{n\pi x}{L}),S_n=sin(\frac{n\pi x}{L})$. \ \ $\Rightarrow$ \ \ $(S_n,S_m)=(C_n,C_m)=L\delta_{mn}$ and $(S_n,C_m)=(C_m,S_n)=0$. \ \ \ \ {\footnotesize $n=1,2,...$, in $[-L,L]$}

\subsection{Fourier Series}

\textbf{Fourier Series}: Periodic function with $T=2L$ \ , \ $f(x)=\frac{a_0}{2}+\sum_{n=1}^{\infty}\left(a_ncos(\frac{n\pi x}{L})+b_nsin(\frac{n\pi x}{L})\right)$ (*) \ \ \ {\scriptsize where $a_0,a_n,b_n$ are Fourier Coefficients.}

\textbf{Euler-Fourier Formulas}: Suppose (*) converges in the interval $[-L,L]$. Then:

\hspace{116pt}$a_n=\frac{1}{L}\int^L_{-L}f(x)cos(\frac{n\pi x}{L})dx$ \ \ \ \ \ \ \ \ $b_n=\frac{1}{L}\int^L_{-L}f(x)sin(\frac{n\pi x}{L})dx$ \ \ \ \ \ \ \ \ $a_0=\frac{1}{L}\int^L_{-L}f(x)dx$ {\scriptsize (Average Value of $f$ over $2L$)}

\textbf{Piecewise Continuous}: {\footnotesize $f(x)$ piecewise continuous on $[a,b]$ \ \ iff \ \ $\exists a=x_0<\cdots<x_n=b$ s.t. $^1f(x)$ continuous $\forall x\in(x_{i-1},x_i)$ $^2 f(x_i^{\pm})=\lim_{x\to x_i^{\pm}}f(x)$ finite.}

\textbf{Fourier Convergence Theorem}: If $^1f,f'$ piecewise continuous on $[-L,L]$,$^2f(x+2L)=f(x)$. \ \ $\Rightarrow$ \ \ Fourier Series of $f$ converges.

$\cdot$ Futhermore, Let Fourier Series of $f$ is $F$. \ \ \ \ $F(x)\to f(x)$, $f(x)$ continuous; $F(x)\to\frac{f(x^+)+f(x^-)}{2}$, $f(x)$ discontinuous.

\textbf{Theorem}: Fourier Series for piecewise continuous functions can be \underline{Integrated term-by-term} to \underline{Convergent} Fourier Seires.

\textbf{Finite Truncation (Partial Sum)}: {\footnotesize Finite Truncation of Fourier Series is: $s_k=\frac{a_0}{2}+\sum_{n=1}^{k}\left(a_ncos(\frac{n\pi x}{L})+b_nsin(\frac{n\pi x}{L})\right)$ \ \ || \ \  \textbf{Error}: $e_N=s_N-f(x)$}

$\cdot$ $n\to\infty$: {\footnotesize $^1$ Continuous points, Fourier approximation of $f(x)$ is better ($|e_n|$ decrease). \ $^2$ Discontinuous points, oscillation of Fourier series get smaller.}

$\cdot$ \textbf{Gibbs Phenomenon}: The Amplitude of the \textit{largest} oscillation (near discontinuous points) does \textit{not decrease}.


\subsection{Other Results on Fourier Series}

\textbf{Extension of a function}: {\scriptsize $f(x)$ know for $x\in[0,L]$ \ \ $\Rightarrow$ \ \ Even Ext.: $h(x)=\{f(x),[0,L]; \ f(-x),(-L,0)$ \ || \ Odd Ext.: $h(x)=\{f(x),(0,L); \ 0,x=\pm L,0; \ -f(-x),(-L,0)$}

\hspace{110pt} {\small Even Extension: $a_n=\frac{2}{L}\int^L_0f(x)cos(\frac{n\pi x}{L})dx,b_n=0$ \ \ \ \ \ \ || \ \ \ \ \ \ Odd Extension: $b_n=\frac{2}{L}\int^L_0f(x)sin(\frac{n\pi x}{L})dx,a_n=0$}

\textbf{Complex Form of Fourier Series}: {\small $f(x)=\sum_{-\infty}^{\infty}c_ne^{\frac{i \ n\pi x}{L}}$} \ \ || \ \ {\footnotesize $c_n=\frac{a_n-ib_n}{2};c_{-n}=\frac{a_n+ib_n}{2};c_0=\frac{a_0}{2},n>0$ \ \ || \ \ $a_0=2c_0;a_n=c_n+c_{-n};b_n=i(c_n-c_{-n})$}

\textbf{Euler-Fourier Formula (Complex)}: {\small If ($\Uparrow$) of Periodic $f,(T=2L)$ converges in $[-L,L]$ \ \ $\Rightarrow$ \ \ $c_n=\frac{1}{2L}\int_{-L}^{L}f(x)e^{-\frac{i \ n\pi x}{L}}$} \ || \ $c_{-n}=\overline{c_n}$ if $f$ real.

\textbf{Parseval's Theorem}: If $^1$ Periodic function $f$; $^2$ $f$ has convergent Fourier Series. 

\hspace{93pt} {\small $\Rightarrow$ \ Square Norm $(f,f)=\int_{-L}^{L}|f(x)|^2dx=2L\sum_{-\infty}^{\infty}|c_n|^2=L\left[\frac{|a_0|^2}{2}+\sum_{n=1}^{\infty}(|a_n|^2+|b_n|^2)\right]$} \ \ \ {\footnotesize ps: 无限维的勾股定理}
